```rust
//! # asterc – Aster Compiler Driver

use clap::Parser;
use std::path::PathBuf;
use aster_compiler::{compile, driver::Config};

#[derive(Parser)]
#[command(name = "asterc", version = "0.1.0", about = "Aster compiler driver")]
struct Cli {
/// Input file
input: PathBuf,

/// Output file
#[arg(short, long)]
output: Option<PathBuf>,

/// Optimization level (0-3)
#[arg(short, long, default_value = "2")]
optimize: u8,

/// Target triple
#[arg(long)]
target: Option<String>,

/// Emit LLVM IR
#[arg(long)]
emit_llvm: bool,

/// Emit assembly
#[arg(long)]
emit_asm: bool,

/// Object file only (no linking)
#[arg(long)]
object_only: bool,

/// Emit MIR (for debugging)
#[arg(long)]
emit_mir: bool,

/// Emit hardware (Verilog)
#[arg(long)]
emit_hdl: bool,

/// No standard library (embedded)
#[arg(long)]
no_std: bool,

/// Verbose output
#[arg(short, long)]
verbose: bool,
}

fn main() -> Result<(), String> {
let cli = Cli::parse();

let output_path = cli.output.unwrap_or_else(|| {
let stem = cli.input.file_stem().unwrap_or_default();
PathBuf::from(stem)
});

let mut config = Config::new(cli.input, output_path);
config.optimization_level = cli.optimize;
config.target_triple = cli.target.unwrap_or_else(|| "native".to_string());
config.emit_ir = cli.emit_llvm;
config.emit_asm = cli.emit_asm;
config.no_std = cli.no_std;

if cli.verbose {
println!("🔧 Compiling with target: {}", config.target_triple);
}

compile(&config)?;

if cli.verbose {

}

Ok(())
```

```rust
let prefix = &line[..position.character as usize];

// Keywords
let keywords = vec![

];

for kw in keywords {

kind: Some(CompletionItemKind::KEYWORD),
detail: Some(format!("Keyword: {}", kw)),
insert_text: Some(kw.to_string()),
..Default::default()
});
}
}

let types = vec!["int", "float", "string", "bool", "char", "unit"];

items.push(CompletionItem {

kind: Some(CompletionItemKind::TYPE_PARAMETER),
detail: Some(format!("Built-in type: {}", ty)),
insert_text: Some(ty.to_string()),
..Default::default()

}

let functions = vec![
```

```rust
use tower_lsp::lsp_types::*;
use std::collections::HashMap;

pub struct DefinitionProvider;

impl DefinitionProvider {
pub fn find_definition(position: Position, source: &str) -> Option<Location> {
// Find the word at the cursor position
let lines: Vec<&str> = source.lines().collect();
if position.line as usize >= lines.len() {
return None;
}

let line = lines[position.line as usize];
let chars: Vec<char> = line.chars().collect();
let mut start = position.character as usize;
let mut end = position.character as usize;

// Find word boundaries
while start > 0 && chars[start - 1].is_ascii_alphanumeric() {
start -= 1;
}
while end < chars.len() && chars[end].is_ascii_alphanumeric() {
end += 1;
}

let word: String = chars[start..end].iter().collect();
if word.is_empty() {
return None;
}

// Search for definition in the source
let mut def_line = 0;
let mut def_col = 0;

for (i, line_text) in lines.iter().enumerate() {
// Look for 'fn <word>' or 'let <word>' or 'struct <word>' or 'enum <word>'
if line_text.contains(&format!("fn {}", word))
|| line_text.contains(&format!("let {}", word))
|| line_text.contains(&format!("struct {}", word))
|| line_text.contains(&format!("enum {}", word))
{
def_line = i;
if let Some(pos) = line_text.find(&word) {
def_col = pos;
break;
}
}
}

Some(Location {
uri: "file:///main.ast".to_string(),
range: Range {
start: Position {
line: def_line as u32,
character: def_col as u32,
},
end: Position {
line: def_line as u32,
character: (def_col + word.len()) as u32,
},
},
})
}
}
```

```rust
if line.contains(&format!("fn {}", word)) {
"Function definition"
} else if line.contains(&format!("let {}", word)) {
"Variable binding"
} else if line.contains(&format!("struct {}", word)) {
"Structure definition"
} else if line.contains(&format!("enum {}", word)) {
"Enumeration definition"
} else {
"Identifier"
}
}
};

Some(Hover {
contents: HoverContents::Scalar(MarkedString::String(content.to_string())),
range: Some(Range {
start: Position {
line: position.line,
character: start as u32,
},
end: Position {
line: position.line,
character: end as u32,
},
}),
})
}
}
```

```rust
//! # Formatting Provider – Full Implementation

use tower_lsp::lsp_types::*;

pub struct FormattingProvider;

impl FormattingProvider {
pub fn format_document(source: &str, options: &FormattingOptions) -> Option<Vec<TextEdit>> {
let formatted = format_code(source, options);

let line_count = source.lines().count();
let end_line = line_count.saturating_sub(1) as u32;
let end_char = source.lines().last().unwrap_or("").len() as u32;

let edits = vec![TextEdit {
range: Range {
start: Position { line: 0, character: 0 },
end: Position { line: end_line, character: end_char },
},
new_text: formatted,
}];

Some(edits)
}
}

fn format_code(source: &str, options: &FormattingOptions) -> String {
let mut result = String::new();
let mut indent = 0;
let tab_size = options.tab_size as usize;
let use_spaces = !options.insert_spaces;

for line in source.lines() {
let trimmed = line.trim();

// Decrease indent for closing braces
if trimmed.starts_with('}') || trimmed.starts_with(')') || trimmed.starts_with(']') {
indent = indent.saturating_sub(1);
}

let indent_str = if use_spaces {
" ".repeat(indent * tab_size)
} else {
"\t".repeat(indent)
};

// Add empty lines for separation
if trimmed.is_empty() {
result.push('\n');
continue;
}

result.push_str(&indent_str);
result.push_str(trimmed);
result.push('\n');

// Increase indent for opening braces
if trimmed.ends_with('{') || trimmed.ends_with('(') || trimmed.ends_with('[') {
indent += 1;
}

// Special: for statements like `if`, `while`, `for`, `match` that end with `{`
// (already handled by above check)
// Also handle `where` and `=>`
if trimmed.contains("=>") || trimmed.contains("where") {
// Don't change indent for these
}
}

result
}
```
