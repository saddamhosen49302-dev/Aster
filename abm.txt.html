//!
//! Supports all popular microcontrollers AND single-board computers.

// Board Definition Structures (Extended for SBC)
// ============================================================

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Board {
pub name: String,
pub manufacturer: String,
pub mcu: String, // MCU or SoC name
pub architecture: Architecture,
pub board_type: BoardType, // NEW: MCU or SBC
pub f_cpu: u32,
pub flash_size: u32,
pub ram_size: u32,
pub eeprom_size: Option<u32>,
pub linker_script: String,
pub hal: String,
pub toolchain: Toolchain,
pub pins: HashMap<String, Pin>,
pub peripherals: Vec<String>,
pub usb_vid: Option<u16>,
pub usb_pid: Option<u16>,
// ---- SBC-specific fields ----
pub os: Option<Vec<String>>, // e.g., ["Raspbian", "Ubuntu", "Android"]
pub gpu: Option<String>, // e.g., "Mali-G52", "Vivante GC7000"
pub wifi_chip: Option<String>, // e.g., "BCM43455", "RTL8822CS"
pub default_distro: Option<String>, // e.g., "Raspberry Pi OS"
pub package_manager: Option<String>, // e.g., "apt", "pacman", "dnf"
pub build_system: Option<String>, // e.g., "buildroot", "yocto", "debootstrap"
pub kernel: Option<String>, // e.g., "linux-6.1", "linux-5.15"
pub vendor_specific: serde_json::Value,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]

Mcu,
Sbc,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum Architecture {
Avr,
Arm,
Xtensa,
RiscV,
Pic,
Mips,
Msp430,
Rl78,
H8,
V850,
Arc,
C166,
Tricore,
PowerPc,
Z80,
X86, // NEW for SBCs
X86_64, // NEW for SBCs
#[serde(other)]
Unknown,
}

#[derive(Debug, Clone, Serialize, Deserialize)]

pub compiler: String,
pub assembler: String,
pub linker: String,
pub objcopy: String,
pub flasher: String,
pub debugger: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Pin {
pub function: String,
pub alt_functions: Vec<String>,
pub analog_channel: Option<u8>,
pub pwm_channel: Option<u8>,
pub default_mode: Option<String>,
}

// ============================================================
// Board Manager
// ============================================================

pub struct BoardManager {
boards_dir: PathBuf,
boards: HashMap<String, Board>,
usb_map: HashMap<(u16, u16), String>,
}

impl BoardManager {
pub fn new() -> Self {
let home = dirs::home_dir().unwrap_or_else(|| PathBuf::from("."));
let boards_dir = home.join(".aster/boards");
if !boards_dir.exists() {
let _ = fs::create_dir_all(&boards_dir);
}

let mut manager = BoardManager {
boards_dir,
boards: HashMap::new(),
usb_map: HashMap::new(),
};
manager.load_builtin_boards();
manager.load_external_boards();
manager.build_usb_map();
manager
}

fn add_sbc(&mut self, board: Board) {
let name = board.name.clone();
if let (Some(vid), Some(pid)) = (board.usb_vid, board.usb_pid) {

// ---- Load external boards ----
fn load_external_boards(&mut self) {
if let Ok(entries) = fs::read_dir(&self.boards_dir) {
for entry in entries.flatten() {
let path = entry.path();
if let Some(ext) = path.extension() {
if ext == "json" || ext == "toml" {
if let Ok(board) = Self::load_board_from_file(&path) {
self.boards.insert(board.name.clone(), board);
}
}
}
}
}
}

fn load_board_from_file(path: &Path) -> Result<Board> {
let content = fs::read_to_string(path)?;
if let Some(ext) = path.extension() {
if ext == "json" {
Ok(serde_json::from_str(&content)?)
} else {
Ok(toml::from_str(&content)?)
}
} else {
Err(anyhow!("Unknown format"))
}
}

fn build_usb_map(&mut self) {
for board in self.boards.values() {
if let (Some(vid), Some(pid)) = (board.usb_vid, board.usb_pid) {
self.usb_map.insert((vid, pid), board.name.clone());
}
}
}

pub fn install_board(&self, name: &str, url: &str) -> Result<()> {
let target_path = self.boards_dir.join(format!("{}.json", name));

if url.starts_with("http") || url.starts_with("git@") {
let temp_dir = tempfile::tempdir()?;
let status = Command::new("git")
.args(["clone", url, temp_dir.path().to_str().unwrap()])
.status()?;

if !status.success() {
return Err(anyhow!("Git clone failed"));
}

for entry in fs::read_dir(temp_dir.path())? {
let entry = entry?;
let path = entry.path();
if let Some(ext) = path.extension() {
if ext == "json" || ext == "toml" {
let content = fs::read_to_string(&path)?;
fs::write(&target_path, content)?;
println!("✅ Board '{}' installed from {}", name, url);
return Ok(());
}
}
}
Err(anyhow!("No board definition found in repository"))
} else {
let response = reqwest::blocking::get(url)?;
let content = response.text()?;
fs::write(&target_path, content)?;
println!("✅ Board '{}' installed from {}", name, url);
Ok(())
}
}

pub fn remove_board(&self, name: &str) -> Result<()> {
let path = self.boards_dir.join(format!("{}.json", name));
if path.exists() {
fs::remove_file(path)?;
println!("✅ Board '{}' removed", name);
Ok(())
} else {
Err(anyhow!("Board '{}' not found", name))
}
}

pub fn to_compiler_config(&self, board: &Board) -> crate::driver::Config {

config.optimize = 2;

config.include_paths.push(
self.boards_dir.join(&board.name).join("include")
);
config.include_paths.push(
self.boards_dir.join(&board.name).join("linux-headers")
);
}
config
}

use clap::{Args, Subcommand};
use aster_compiler::board_manager::{BoardManager, BoardType};
use anyhow::Result;

#[derive(Args)]
pub struct BoardArgs {
#[command(subcommand)]
pub command: BoardCommands,
}

#[derive(Subcommand)]
pub enum BoardCommands {
List {
#[arg(long)]
sbc: bool,
#[arg(long)]
mcu: bool,
#[arg(long)]
all: bool,
},
Info { name: String },
Add { name: String, url: String },
Remove { name: String },
Detect,
}

pub fn handle_board(args: BoardArgs) -> Result<()> {
let manager = BoardManager::new();

match args.command {
BoardCommands::List { sbc, mcu, all } => {
let boards = if all {
manager.list_boards()
} else if sbc {
manager.list_by_type(BoardType::Sbc)
} else if mcu {
manager.list_by_type(BoardType::Mcu)
} else {
manager.list_boards()
};

println!("📋 Boards:");
for board in boards {
let board_type = if board.board_type == BoardType::Sbc { "SBC" } else { "MCU" };
let extra = if let Some(os) = &board.os {
format!(", OS: {}", os.join("/"))
} else {
String::new()
};
println!(
" - {} ({} by {}, {} MHz{})",
board.name, board.mcu, board.manufacturer, board.f_cpu / 1_000_000, extra
);
}
Ok(())
}
BoardCommands::Info { name } => {
if let Some(board) = manager.get_board(&name) {
println!("📄 Board: {}", board.name);
println!(" Type: {:?}", board.board_type);
println!(" Manufacturer: {}", board.manufacturer);
println!(" SoC/MCU: {}", board.mcu);
println!(" Architecture: {:?}", board.architecture);
println!(" CPU Freq: {} MHz", board.f_cpu / 1_000_000);
if board.board_type == BoardType::Sbc {
if let Some(os) = &board.os {
println!(" OS Support: {}", os.join(", "));
}
if let Some(gpu) = &board.gpu {
println!(" GPU: {}", gpu);
}
if let Some(wifi) = &board.wifi_chip {
println!(" WiFi: {}", wifi);
}
if let Some(dist) = &board.default_distro {
println!(" Default Distro: {}", dist);
}
if let Some(pm) = &board.package_manager {
println!(" Package Manager: {}", pm);
}
if let Some(kernel) = &board.kernel {
println!(" Kernel: {}", kernel);
}
} else {
println!(" Flash: {} KB", board.flash_size / 1024);
println!(" RAM: {} KB", board.ram_size / 1024);
}
println!(" Peripherals: {:?}", board.peripherals);
Ok(())
} else {
anyhow::bail!("Board '{}' not found", name)
}
}
BoardCommands::Add { name, url } => {
manager.install_board(&name, &url)?;
Ok(())
}
BoardCommands::Remove { name } => {
manager.remove_board(&name)?;
Ok(())
}
BoardCommands::Detect => {
println!("🔍 Scanning USB devices...");
// Real implementation via udev/libusb
let sample_devices = vec![
(0x2341, 0x0043, "Arduino Uno"),
(0x2E8A, 0x0003, "Raspberry Pi Pico"),
(0x0483, 0x3748, "STM32F103"),
(0x1D6B, 0x0002, "BeagleBone Black"),
(0x16C0, 0x0483, "Teensy 4.0"),
];
for (vid, pid, name) in sample_devices {
if let Some(board) = manager.detect_by_usb(vid, pid) {
println!(" ✅ Found: {} (VID:{:04X} PID:{:04X})", board.name, vid, pid);
} else {
println!(" ⚠️ Device {} not in database", name);
}
}
Ok(())
}
}
}
