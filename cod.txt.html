```rust
Unary(UnaryOp, Operand),

}

#[derive(Debug, Clone)]
pub enum Operand {
Copy(Place),
Move(Place),
Constant(Constant),
}

#[derive(Debug, Clone)]

Int(i32),
Bool(bool),
}

#[derive(Debug, Clone)]

Add, Sub, Mul, Div,

}

#[derive(Debug, Clone)]

Neg, Not,
}

#[derive(Debug, Clone)]

pub kind: TerminatorKind,
}

#[derive(Debug, Clone)]

Return,
Goto(usize),
SwitchInt(Operand, Vec<(Constant, usize)>, Option<usize>),
Unreachable,
}
}

use codegen::Codegen;
```

```rust
/// Standard (host) compilation via LLVM
pub fn generate(&mut self, bodies: &[MirBody], output: &Output) -> Result<(), String> {
let context = inkwell::context::Context::create();
let mut llvm = LlvmCodegen::new(&context, "aster_module");
llvm.generate(bodies)?;
llvm.verify()?;
if output.emit_ir {
llvm.write_to_file(&output.ir_path(), &self.config)?;
}
if output.emit_asm {
llvm.compile_to_asm(&output.asm_path(), &self.config)?;
}
llvm.compile_to_object(&output.object_path(), &self.config)?;
Ok(())
}

/// Embedded / MCU compilation (with linker script & startup)
pub fn generate_embedded(&mut self, bodies: &[MirBody], output: &Output) -> Result<(), String> {
let mut mcu = McuCodegen::new(&self.config);
mcu.analyze(bodies);
let out_dir = output.path.parent().unwrap_or(Path::new("."));
mcu.emit(out_dir)?;

let context = inkwell::context::Context::create();
let mut llvm = LlvmCodegen::new(&context, "firmware");
llvm.generate(bodies)?;
llvm.verify()?;
llvm.compile_to_object(&out_dir.join("firmware.elf"), &self.config)?;
Ok(())
}
```

```rust
//! LLVM backend with target machine support

use inkwell::context::Context;
use inkwell::module::Module;
use inkwell::builder::Builder;
use inkwell::values::{PointerValue, FunctionValue, BasicValueEnum};
use inkwell::targets::{Target, TargetMachine, CodeModel, FileType, RelocMode};
use crate::mir::{MirBody, Statement, TerminatorKind, Rvalue, Operand, Place, Constant, BinaryOp, UnaryOp};
use crate::driver::Config;
use std::collections::HashMap;
use std::path::Path;

pub struct LlvmCodegen<'ctx> {
context: &'ctx Context,
module: Module<'ctx>,
builder: Builder<'ctx>,
locals: HashMap<Place, PointerValue<'ctx>>,
functions: HashMap<String, FunctionValue<'ctx>>,
}

impl<'ctx> LlvmCodegen<'ctx> {
pub fn new(context: &'ctx Context, name: &str) -> Self {
LlvmCodegen {
context,
module: context.create_module(name),
builder: context.create_builder(),
locals: HashMap::new(),
functions: HashMap::new(),
}
}

pub fn generate(&mut self, bodies: &[MirBody]) -> Result<(), String> {
for body in bodies {
self.declare_function(body)?;
}
for body in bodies {
self.define_function(body)?;
}
Ok(())
}

fn declare_function(&mut self, body: &MirBody) -> Result<(), String> {
let i32_type = self.context.i32_type();
let void_type = self.context.void_type();
let ret_type = if body.basic_blocks.last().map(|b| matches!(b.terminator.kind, TerminatorKind::Return)).unwrap_or(false) {
i32_type.as_basic_type_enum()
} else {
void_type.as_basic_type_enum()
};
let param_types = vec![i32_type.into(); body.arg_count];
let fn_type = i32_type.fn_type(&param_types, false);
let fn_val = self.module.add_function(&body.name, fn_type, None);
self.functions.insert(body.name.clone(), fn_val);
Ok(())
}

fn define_function(&mut self, body: &MirBody) -> Result<(), String> {
let fn_val = self.functions.get(&body.name).unwrap();
let entry = self.context.append_basic_block(*fn_val, "entry");
self.builder.position_at_end(entry);
let i32_type = self.context.i32_type();

for (i, arg) in fn_val.get_param_iter().enumerate() {
let alloca = self.builder.build_alloca(i32_type, &format!("arg_{}", i))?;
self.builder.build_store(alloca, arg)?;
self.locals.insert(Place { name: format!("arg_{}", i), index: Some(i) }, alloca);
}

for (idx, block) in body.basic_blocks.iter().enumerate() {
if idx > 0 {
self.builder.position_at_end(self.context.append_basic_block(*fn_val, &format!("bb{}", idx)));
}
for stmt in &block.statements {
self.compile_statement(stmt)?;
}
self.compile_terminator(&block.terminator)?;
}
Ok(())
}

fn compile_statement(&mut self, stmt: &Statement) -> Result<(), String> {
match stmt {
Statement::Assign(p, rv) => {
let val = self.compile_rvalue(rv)?;
let ptr = self.get_or_create_local(p)?;
self.builder.build_store(ptr, val)?;
}
_ => {}
}
Ok(())
}

fn compile_rvalue(&mut self, rv: &Rvalue) -> Result<BasicValueEnum<'ctx>, String> {
match rv {
Rvalue::Binary(op, l, r) => {
let lv = self.compile_operand(l)?.into_int_value();
let rv = self.compile_operand(r)?.into_int_value();
let result = match op {
BinaryOp::Add => self.builder.build_int_add(lv, rv, "add")?,
BinaryOp::Sub => self.builder.build_int_sub(lv, rv, "sub")?,
_ => return Err("Unsupported binary op".into()),
};
Ok(result.as_basic_value_enum())
}
_ => Ok(self.context.i32_type().const_zero().as_basic_value_enum()),
}
}

fn compile_operand(&mut self, op: &Operand) -> Result<BasicValueEnum<'ctx>, String> {
match op {
Operand::Constant(c) => match c {
Constant::Int(v) => Ok(self.context.i32_type().const_int(*v as u64, false).as_basic_value_enum()),
_ => Ok(self.context.i32_type().const_zero().as_basic_value_enum()),
}
Operand::Move(p) | Operand::Copy(p) => {
let ptr = self.locals.get(p).ok_or("Place not found")?;
Ok(self.builder.build_load(self.context.i32_type(), *ptr, "load")?)
}
}
}

fn compile_terminator(&mut self, term: &crate::mir::Terminator) -> Result<(), String> {
match &term.kind {
TerminatorKind::Return => {
self.builder.build_return(Some(&self.context.i32_type().const_zero()))?;
}
_ => {}
}
Ok(())
}

fn get_or_create_local(&mut self, p: &Place) -> Result<PointerValue<'ctx>, String> {
if let Some(ptr) = self.locals.get(p) {
return Ok(*ptr);
}
let alloca = self.builder.build_alloca(self.context.i32_type(), &p.name)?;
self.locals.insert(p.clone(), alloca);
Ok(alloca)
}

pub fn write_to_file(&self, path: &Path, _config: &Config) -> Result<(), String> {
self.module.print_to_file(path).map_err(|e| e.to_string())
}

fn create_tm(&self, config: &Config) -> Result<TargetMachine, String> {
Target::initialize_all(&inkwell::targets::InitializationConfig::default());
let target = Target::from_triple(&config.target_triple).map_err(|e| e.to_string())?;
target.create_target_machine(
&config.target_triple,
"generic",
"",
CodeModel::Default,
RelocMode::Default,
inkwell::OptimizationLevel::Default,
)
.ok_or("Failed to create target machine".into())
}

pub fn compile_to_object(&self, path: &Path, config: &Config) -> Result<(), String> {
let tm = self.create_tm(config)?;
tm.write_to_file(&self.module, FileType::Object, path).map_err(|e| e.to_string())
}

pub fn compile_to_asm(&self, path: &Path, config: &Config) -> Result<(), String> {
let tm = self.create_tm(config)?;
tm.write_to_file(&self.module, FileType::Assembly, path).map_err(|e| e.to_string())
}

pub fn verify(&self) -> Result<(), String> {
self.module.verify().map_err(|e| format!("{:?}", e))
}
}
```

```rust
//! MCU / embedded code generator (Raspberry Pi, RP2040, AVR)

use crate::mir::MirBody;
use crate::driver::Config;
use std::collections::HashSet;
use std::path::Path;

pub struct McuCodegen {
config: Config,
interrupt_handlers: HashSet<String>,
entry_point: Option<String>,
}

impl McuCodegen {
pub fn new(config: &Config) -> Self {
McuCodegen {
config: config.clone(),
interrupt_handlers: HashSet::new(),
entry_point: None,
}
}

pub fn analyze(&mut self, bodies: &[MirBody]) {
for body in bodies {
if body.attrs.iter().any(|a| a == "#[entry]") {
self.entry_point = Some(body.name.clone());
}
if body.attrs.iter().any(|a| a.starts_with("#[interrupt]")) {
self.interrupt_handlers.insert(body.name.clone());
}
}
}

pub fn generate_linker_script(&self) -> Result<String, String> {
let f = &self.config.mcu_family;
let (ram, ram_sz, flash, flash_sz) = match f.as_str() {
"raspberry_pi_arm64" => ("0x80000", "0x1000000", "0x80000", "0x1000000"),
"raspberry_pi_armv7" => ("0x8000", "0x1000000", "0x8000", "0x1000000"),
"rp2040" => ("0x20000000", "0x40000", "0x10000000", "0x200000"),
_ => ("0x20000000", "0x10000", "0x08000000", "0x100000"),
};
Ok(format!(
"MEMORY {{ FLASH (rx) : ORIGIN = {}, LENGTH = {}\n RAM (rwx) : ORIGIN = {}, LENGTH = {} }}\n\
SECTIONS {{ .text : {{ *(.text) }} > FLASH .data : {{ *(.data) }} > RAM .bss : {{ *(.bss) }} > RAM }}",
flash, flash_sz, ram, ram_sz
))
}

pub fn generate_startup(&self) -> Result<String, String> {
let f = &self.config.mcu_family;
match f.as_str() {
"raspberry_pi_arm64" => Ok(".section .text\n.globl _start\n_start: adr x0, _estack; mov sp, x0; bl main; b .".into()),
"rp2040" => Ok(".section .vector_table\n.word _estack\n.word _start\n_start: bl main; b .".into()),
_ => Ok(".globl _start\n_start: bl main; b .".into()),
}
}

pub fn emit(&self, out_dir: &Path) -> Result<(), String> {
std::fs::write(out_dir.join("memory.ld"), self.generate_linker_script()?)?;
std::fs::write(out_dir.join("startup.s"), self.generate_startup()?)?;
Ok(())
}
}
```
