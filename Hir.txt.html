HirType::Bool => write!(f, "bool"),
HirType::Int => write!(f, "int"),
HirType::Float => write!(f, "float"),

//! # AST → HIR Lowering
//!
//! Transforms the typed AST into HIR using semantic context.

use crate::ast::*;
use crate::sema::SemaContext;
use super::*;
use super::context::HirContext;
use std::collections::HashMap;

pub struct Lowering<'a> {
sema_ctx: &'a SemaContext,
hir_ctx: HirContext,
// SEMA থেকে প্রাপ্ত টাইপ ম্যাপ (ধরে নিচ্ছি SemaContext-এ একটি ফিল্ড আছে)
type_map: HashMap<String, HirType>,
// স্থানীয় স্কোপ স্ট্যাক (ভেরিয়েবল নাম -> টাইপ)
locals: Vec<HashMap<String, HirType>>,
}

impl<'a> Lowering<'a> {
pub fn new(sema_ctx: &'a SemaContext) -> Self {
// এখানে আসল SEMA থেকে টাইপ ম্যাপ বের করতে হবে।
// উদাহরণস্বরূপ: sema_ctx.get_type_map()
let dummy_type_map = HashMap::new();
Lowering {
sema_ctx,
hir_ctx: HirContext::new(),
type_map: dummy_type_map,
locals: Vec::new(),
}
}

// ----- স্কোপ ম্যানেজমেন্ট -----
fn push_scope(&mut self) {
self.locals.push(HashMap::new());
}

fn pop_scope(&mut self) {
self.locals.pop();
}

fn declare_var(&mut self, name: &str, ty: HirType) {
if let Some(scope) = self.locals.last_mut() {
scope.insert(name.to_string(), ty);
}
}

fn lookup_var(&self, name: &str) -> Option<HirType> {
for scope in self.locals.iter().rev() {
if let Some(ty) = scope.get(name) {
return Some(ty.clone());
}
}
// গ্লোবাল ম্যাপেও খুঁজতে পারেন
self.type_map.get(name).cloned()
}

// ----- পাবলিক এন্ট্রি -----
pub fn lower_file(&mut self, file: &File) -> Result<HirFile, Vec<String>> {
let mut items = Vec::new();
for item in &file.items {
match self.lower_item(item) {
Ok(hi) => items.push(hi),
Err(e) => return Err(vec![e]),
}
}
Ok(HirFile::new(items, self.hir_ctx.clone()))
}

// ----- আইটেম লোয়ারিং -----
fn lower_item(&mut self, item: &Item) -> Result<HirItem, String> {
match item {
Item::Fn(sig) => {
self.push_scope();
// আর্গুমেন্ট ঘোষণা
let mut args = Vec::new();
for (pat, ty) in &sig.args {
if let Pat::Ident(name, _, _, _) = pat {
let hir_ty = self.lower_ty(ty);
self.declare_var(name, hir_ty.clone());
args.push((name.clone(), hir_ty));
} else {
return Err("function argument must be identifier".to_string());
}
}
let body = self.lower_block(&sig.body)?;
self.pop_scope();
Ok(HirItem::Fn(HirFn {
name: sig.name.clone(),
generics: sig.generics.clone(),
args,
ret_ty: self.lower_ty(&sig.ret_ty),
body,
span: sig.span,
}))
}
Item::Struct(name, fields, generics, _, span) => {
let hir_fields = fields
.iter()
.map(|f| HirStructField {
name: f.name.clone(),
ty: self.lower_ty(&f.ty),
span: f.span,
})
.collect();
Ok(HirItem::Struct(HirStruct {
name: name.clone(),
generics: generics.clone(),
fields: hir_fields,
span: *span,
}))
}
Item::Enum(name, variants, generics, _, span) => {
let hir_variants = variants
.iter()
.map(|v| HirEnumVariant {
name: v.name.clone(),
fields: v.fields.iter().map(|t| self.lower_ty(t)).collect(),
span: v.span,
})
.collect();
Ok(HirItem::Enum(HirEnum {
name: name.clone(),
generics: generics.clone(),
variants: hir_variants,
span: *span,
}))
}
Item::Trait(name, methods, generics, _, span) => {
let hir_methods = methods
.iter()
.map(|m| self.lower_fn_sig(m))
.collect::<Result<Vec<_>, _>>()?;
Ok(HirItem::Trait(HirTrait {
name: name.clone(),
generics: generics.clone(),
methods: hir_methods,
span: *span,
}))
}
Item::Impl(ty, methods, generics, trait_name, _, span) => {
let hir_methods = methods
.iter()
.map(|m| self.lower_fn_sig(m))
.collect::<Result<Vec<_>, _>>()?;
Ok(HirItem::Impl(HirImpl {
trait_name: trait_name.clone(),
ty: self.lower_ty(ty),
methods: hir_methods,
span: *span,
}))
}
Item::Mod(name, items, _) => {
let mut hir_items = Vec::new();
for sub in items {
hir_items.push(self.lower_item(sub)?);
}
Ok(HirItem::Mod(name.clone(), hir_items))
}
_ => Err(format!("unsupported item: {:?}", item)),
}
}

fn lower_fn_sig(&mut self, sig: &FnSig) -> Result<HirFn, String> {
// নোট: ট্রেইট বা ইমপ্লের মেথডের বডি সাধারণত লোয়ার করা হয় না (শুধু সিগনেচার)
// কিন্তু আমরা এখানে ডামি বডি রেখেছি।
let args = sig
.args
.iter()
.map(|(pat, ty)| {
if let Pat::Ident(name, _, _, _) = pat {
Ok((name.clone(), self.lower_ty(ty)))
} else {
Err("argument must be identifier".to_string())
}
})
.collect::<Result<Vec<_>, _>>()?;
// ডামি বডি (যেহেতু ট্রেইট/ইমপ্লে বডি থাকে না)
let dummy_block = HirBlock { stmts: Vec::new(), span: sig.span };
Ok(HirFn {
name: sig.name.clone(),
generics: sig.generics.clone(),
args,
ret_ty: self.lower_ty(&sig.ret_ty),
body: dummy_block,
span: sig.span,
})
}

// ----- ব্লক ও স্টেটমেন্ট -----
fn lower_block(&mut self, block: &Block) -> Result<HirBlock, String> {
self.push_scope();
let mut stmts = Vec::new();
for stmt in &block.stmts {
stmts.push(self.lower_stmt(stmt)?);
}
self.pop_scope();
Ok(HirBlock {
stmts,
span: block.span,
})
}

fn lower_stmt(&mut self, stmt: &Stmt) -> Result<HirStmt, String> {
match stmt {
Stmt::Let(pat, ty_opt, expr, span) => {
let hir_pat = self.lower_pat(pat)?;
let hir_expr = self.lower_expr(expr)?;
// প্যাটার্ন ভেরিয়েবল ডিক্লেয়ার (সিম্পলিফাইড)
if let HirPattern::Variable(name, ty) = &hir_pat {
self.declare_var(name, ty.clone());
}
Ok(HirStmt::Let(hir_pat, Some(hir_expr), *span))
}
Stmt::Expr(expr, _) => Ok(HirStmt::Expr(self.lower_expr(expr)?)),
Stmt::Semi(expr, _) => Ok(HirStmt::Semi(self.lower_expr(expr)?)),
Stmt::Return(expr_opt, span) => {
let hir_expr = match expr_opt {
Some(e) => Some(self.lower_expr(e)?),
None => None,
};
Ok(HirStmt::Return(hir_expr, *span))
}
Stmt::If(cond, then_b, else_b, span) => {
let cond = self.lower_expr(cond)?;
let then_block = self.lower_block(then_b)?;
let else_block = match else_b {
Some(b) => Some(self.lower_block(b)?),
None => None,
};
Ok(HirStmt::If(cond, then_block, else_block, *span))
}
Stmt::While(cond, body, span) => {
let cond = self.lower_expr(cond)?;
let body = self.lower_block(body)?;
Ok(HirStmt::While(cond, body, *span))
}
Stmt::For(pat, iter, body, span) => {
let pat = self.lower_pat(pat)?;
let iter = self.lower_expr(iter)?;
let body = self.lower_block(body)?;
Ok(HirStmt::For(pat, iter, body, *span))
}
Stmt::Match(expr, arms, span) => {
let expr = self.lower_expr(expr)?;
let mut hir_arms = Vec::new();
for arm in arms {
let pat = self.lower_pat(&arm.pat)?;
let guard = match &arm.guard {
Some(g) => Some(self.lower_expr(g)?),
None => None,
};
let body = self.lower_block(&arm.body)?;
hir_arms.push(HirArm {
pat,
guard,
body,
span: arm.span,
});
}
Ok(HirStmt::Match(expr, hir_arms, *span))
}
Stmt::Block(block, _) => {
let hir_block = self.lower_block(block)?;
Ok(HirStmt::Block(hir_block))
}
Stmt::Break(expr_opt, span) => Ok(HirStmt::Break(expr_opt.as_ref().map(|e| self.lower_expr(e).ok()).flatten(), *span)),
Stmt::Continue(span) => Ok(HirStmt::Continue(*span)),
_ => Ok(HirStmt::Empty),
}
}

// ----- এক্সপ্রেশন লোয়ারিং (সম্পূর্ণ) -----
fn lower_expr(&mut self, expr: &Expr) -> Result<HirExpr, String> {
match expr {
Expr::IntLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Int(*val)),
ty: HirType::Int,
span: *span,
}),
Expr::FloatLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Float(*val)),
ty: HirType::Float,
span: *span,
}),
Expr::StringLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::String(val.clone())),
ty: HirType::String,
span: *span,
}),
Expr::BoolLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Bool(*val)),
ty: HirType::Bool,
span: *span,
}),
Expr::CharLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Char(*val)),
ty: HirType::Char,
span: *span,
}),
Expr::UnitLit(span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Unit),
ty: HirType::Unit,
span: *span,
}),
Expr::Variable(name, span) => {
let ty = self.lookup_var(name).unwrap_or(HirType::Error);
Ok(HirExpr {
kind: HirExprKind::Variable(name.clone()),
ty,
span: *span,
})
}
Expr::Unary(op, e, span) => {
let e = Box::new(self.lower_expr(e)?);
let hir_op = match op {
UnaryOp::Neg => HirUnaryOp::Neg,
UnaryOp::Not => HirUnaryOp::Not,
UnaryOp::Deref => HirUnaryOp::Deref,
};
Ok(HirExpr {
kind: HirExprKind::Unary(hir_op, e.clone()),
ty: e.ty.clone(),
span: *span,
})
}
Expr::Binary(l, op, r, span) => {
let left = Box::new(self.lower_expr(l)?);
let right = Box::new(self.lower_expr(r)?);
let hir_op = match op {
BinaryOp::Add => HirBinaryOp::Add,
BinaryOp::Sub => HirBinaryOp::Sub,
BinaryOp::Mul => HirBinaryOp::Mul,
BinaryOp::Div => HirBinaryOp::Div,
BinaryOp::Mod => HirBinaryOp::Mod,
BinaryOp::Eq => HirBinaryOp::Eq,
BinaryOp::Ne => HirBinaryOp::Ne,
BinaryOp::Lt => HirBinaryOp::Lt,
BinaryOp::Le => HirBinaryOp::Le,
BinaryOp::Gt => HirBinaryOp::Gt,

};
let ty = left.ty.clone(); // সরলীকৃত

//! # HIR Visitor Trait
//!
//! Enables traversal and transformation of HIR.

use super::*;

pub type VisResult = Result<(), String>;

pub trait HirVisitor {
fn visit_file(&mut self, file: &HirFile) -> VisResult {
for item in &file.items {
self.visit_item(item)?;
}
Ok(())
}

fn visit_item(&mut self, item: &HirItem) -> VisResult {
match item {
HirItem::Fn(f) => self.visit_fn(f),
HirItem::Struct(s) => self.visit_struct(s),
HirItem::Enum(e) => self.visit_enum(e),
HirItem::Trait(t) => self.visit_trait(t),
HirItem::Impl(i) => self.visit_impl(i),
HirItem::Mod(_, items) => {
for sub in items {
self.visit_item(sub)?;
}
Ok(())
}
}
}

fn visit_fn(&mut self, f: &HirFn) -> VisResult {
self.visit_block(&f.body)
}
fn visit_struct(&mut self, _s: &HirStruct) -> VisResult { Ok(()) }
fn visit_enum(&mut self, _e: &HirEnum) -> VisResult { Ok(()) }
fn visit_trait(&mut self, _t: &HirTrait) -> VisResult { Ok(()) }
fn visit_impl(&mut self, _i: &HirImpl) -> VisResult { Ok(()) }

fn visit_block(&mut self, block: &HirBlock) -> VisResult {
for stmt in &block.stmts {
self.visit_stmt(stmt)?;
}
Ok(())
}

fn visit_stmt(&mut self, stmt: &HirStmt) -> VisResult {
match stmt {
HirStmt::Let(_, Some(expr), _) => self.visit_expr(expr),
HirStmt::Let(_, None, _) => Ok(()),
HirStmt::Expr(expr) => self.visit_expr(expr),
HirStmt::Semi(expr) => self.visit_expr(expr),
HirStmt::If(cond, then_block, else_block, _) => {
self.visit_expr(cond)?;
self.visit_block(then_block)?;
if let Some(b) = else_block {
self.visit_block(b)?;
}
Ok(())
}

self.visit_expr(cond)?;
self.visit_block(body)
}
HirStmt::For(_, iter, body, _) => {
self.visit_expr(iter)?;
self.visit_block(body)
}
HirStmt::Match(expr, arms, _) => {
self.visit_expr(expr)?;
for arm in arms {
self.visit_block(&arm.body)?;
}
Ok(())
}
HirStmt::Return(Some(expr), _) => self.visit_expr(expr),
HirStmt::Return(None, _) => Ok(()),
HirStmt::Block(block) => self.visit_block(block),
_ => Ok(()),
}
}

HirExprKind::Literal(_) => Ok(()),
HirExprKind::Variable(_) => Ok(()),
HirExprKind::Unary(_, e) => self.visit_expr(e),
HirExprKind::Binary(l, _, r) => {
self.visit_expr(l)?;
self.visit_expr(r)

HirExprKind::Call(f, args) => {
self.visit_expr(f)?;
for arg in args {
self.visit_expr(arg)?;
}
Ok(())
}
HirExprKind::Block(b) => self.visit_block(b),
HirExprKind::If(cond, then_b, else_e) => {
self.visit_expr(cond)?;
self.visit_block(then_b)?;
if let Some(e) = else_e {
self.visit_expr(e)?;
}
Ok(())
}
HirExprKind::Loop(b) => self.visit_block(b),
HirExprKind::While(cond, b) => {
self.visit_expr(cond)?;
self.visit_block(b)
}
HirExprKind::For(_, iter, b) => {
self.visit_expr(iter)?;
self.visit_block(b)
}
HirExprKind::Match(expr, arms) => {
self.visit_expr(expr)?;
for arm in arms {

self.visit_expr(arr)?;
self.visit_expr(idx)

HirExprKind::Break(Some(e)) => self.visit_expr(e),
HirExprKind::Break(None) => Ok(()),
HirExprKind::Continue => Ok(()),
HirExprKind::Return(Some(e)) => self.visit_expr(e),
HirExprKind::Return(None) => Ok(()),
HirExprKind::Assign(l, r) => {
self.visit_expr(l)?;
self.visit_expr(r)
}
HirExprKind::AddrOf(e) | HirExprKind::Deref(e) => self.visit_expr(e),
}
}

//! # HIR Pretty Printer

}

impl HirPrettyPrinter {
pub fn new() -> Self {
HirPrettyPrinter {
indent: 0,

}
}

pub fn print(&mut self, file: &HirFile) -> String {
self.output.clear();
self.print_file(file);
self.output.clone()
}

" ".repeat(self.indent)
}

fn print_file(&mut self, file: &HirFile) {
for item in &file.items {
self.print_item(item);
}
}

HirItem::Struct(s) => self.print_struct(s),
HirItem::Enum(e) => self.print_enum(e),
HirItem::Trait(t) => self.print_trait(t),
HirItem::Impl(i) => self.print_impl(i),
HirItem::Mod(name, items) => {
writeln!(self.output, "{}mod {} {{", self.indent_str(), name).unwrap();
self.indent += 1;
for sub in items {
self.print_item(sub);
}
self.indent -= 1;
writeln!(self.output, "{}}}", self.indent_str()).unwrap();
}
}
}

}

self.output,

}

}
HirStmt::Block(block) => {

}

self.expr_to_string(iter)

HirExprKind::Match(expr, _) => format!("match {} {{ ... }}", self.expr_to_string(expr)),

es.iter().map(|e| self.expr_to_string(e)).collect::<Vec<_>>().join(", ")
),
HirExprKind::Index(arr, idx) => format!("{}[{}]", self.expr_to_string(arr), self.expr_to_string(idx)),
HirExprKind::Break(Some(e)) => format!("break {}", self.expr_to_string(e)),

HirExprKind::AddrOf(e) => format!("&{}", self.expr_to_string(e)),
HirExprKind::Deref(e) => format!("*{}", self.expr_to_string(e)),
}
}

fn pat_to_string(&self, pat: &HirPattern) -> String {

HirPattern::Variable(name, _) => name.clone(),
HirPattern::Literal(lit) => match lit {

HirLiteral::String(s) => format!("\"{}\"", s),
HirLiteral::Bool(b) => b.to_string(),

},
HirPattern::Tuple(pats) => format!(
"({})",
pats.iter().map(|p| self.pat_to_string(p)).collect::<Vec<_>>().join(", ")
),
HirPattern::Struct(name, fields) => format!(
"{} {{ {} }}",
name,
fields.iter().map(|(k, v)| format!("{}: {}", k, self.pat_to_string(v))).collect::<Vec<_>>().join(", ")
),
HirPattern::Enum(name, pats) => format!(
"{}({})",
name,
pats.iter().map(|p| self.pat_to_string(p)).collect::<Vec<_>>().join(", ")
),
HirPattern::Or(pats) => format!(
"({})",
pats.iter().map(|p| self.pat_to_string(p)).collect::<Vec<_>>().join(" | ")
),
}
}

fn lower_stmt(&mut self, stmt: &Stmt) -> Result<HirStmt, String> {

let hir_expr = self.lower_expr(expr)?;
if let HirPattern::Variable(name, ty) = &hir_pat {
self.declare_var(name, ty.clone());
}
Ok(HirStmt::Let(hir_pat, Some(hir_expr), *span))
}
Stmt::Expr(expr, _) => Ok(HirStmt::Expr(self.lower_expr(expr)?)),
Stmt::Semi(expr, _) => Ok(HirStmt::Semi(self.lower_expr(expr)?)),
Stmt::Return(expr_opt, span) => {
let hir_expr = match expr_opt {
Some(e) => Some(self.lower_expr(e)?),
None => None,
};
Ok(HirStmt::Return(hir_expr, *span))
}
Stmt::If(cond, then_b, else_b, span) => {
let cond = self.lower_expr(cond)?;
let then_block = self.lower_block(then_b)?;
let else_block = match else_b {
Some(b) => Some(self.lower_block(b)?),
None => None,
};
Ok(HirStmt::If(cond, then_block, else_block, *span))
}
Stmt::While(cond, body, span) => {
let cond = self.lower_expr(cond)?;
let body = self.lower_block(body)?;
Ok(HirStmt::While(cond, body, *span))
}
Stmt::For(pat, iter, body, span) => {
let pat = self.lower_pat(pat)?;
let iter = self.lower_expr(iter)?;
let body = self.lower_block(body)?;
Ok(HirStmt::For(pat, iter, body, *span))
}
Stmt::Loop(label, body, span) => {
// HIR-এ লুপ এক্সপ্রেশন নেই, তাই আমরা ব্লক হিসেবে নিই
let body = self.lower_block(body)?;
// লুপ স্টেটমেন্টকে ব্লকে রূপান্তর (পরে ট্রান্সফর্মেশনে ব্যবহার)
Ok(HirStmt::Block(body))
}
Stmt::Break(expr_opt, span) => {
let hir_expr = match expr_opt {
Some(e) => Some(self.lower_expr(e)?),
None => None,
};
Ok(HirStmt::Break(hir_expr, *span))
}
Stmt::Continue(span) => Ok(HirStmt::Continue(*span)),
Stmt::Match(expr, arms, span) => {
let expr = self.lower_expr(expr)?;
let mut hir_arms = Vec::new();
for arm in arms {
let pat = self.lower_pat(&arm.pat)?;
let guard = match &arm.guard {
Some(g) => Some(self.lower_expr(g)?),
None => None,
};
let body = self.lower_block(&arm.body)?;
hir_arms.push(HirArm {
pat,
guard,
body,
span: arm.span,
});
}
Ok(HirStmt::Match(expr, hir_arms, *span))
}
Stmt::Try(try_b, catches, _) => {

let try_block = self.lower_block(try_b)?;
Ok(HirStmt::Block(try_block))
}
Stmt::Block(block, _) => {

}

Stmt::HwAssign(_, _, _) => Ok(HirStmt::Empty),
Stmt::HwAlways(_, _, _) => Ok(HirStmt::Empty),
Stmt::HwAssert(_, _) => Ok(HirStmt::Empty),
_ => Ok(HirStmt::Empty),
}
}

// ----- এক্সপ্রেশন লোয়ারিং (শুধু AST-তে থাকা) -----
fn lower_expr(&mut self, expr: &Expr) -> Result<HirExpr, String> {
match expr {
Expr::IntLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Int(val.parse().unwrap_or(0))),
ty: HirType::Int,
span: *span,
}),
Expr::FloatLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Float(val.parse().unwrap_or(0.0))),
ty: HirType::Float,
span: *span,
}),
Expr::StringLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::String(val.clone())),
ty: HirType::String,
span: *span,
}),
Expr::BoolLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Bool(*val)),
ty: HirType::Bool,
span: *span,
}),
Expr::CharLit(val, span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Char(*val)),
ty: HirType::Char,
span: *span,
}),
Expr::Variable(name, span) => {
let ty = self.lookup_var(name).unwrap_or(HirType::Error);
Ok(HirExpr {
kind: HirExprKind::Variable(name.clone()),
ty,
span: *span,
})
}
Expr::Unary(op, e, span) => {
let e = Box::new(self.lower_expr(e)?);
let hir_op = match op {
UnaryOp::Neg => HirUnaryOp::Neg,
UnaryOp::Not => HirUnaryOp::Not,
UnaryOp::Deref => HirUnaryOp::Deref,
};
Ok(HirExpr {
kind: HirExprKind::Unary(hir_op, e.clone()),
ty: e.ty.clone(),
span: *span,
})
}
Expr::Binary(l, op, r, span) => {
let left = Box::new(self.lower_expr(l)?);
let right = Box::new(self.lower_expr(r)?);
let hir_op = match op {
BinaryOp::Add => HirBinaryOp::Add,
BinaryOp::Sub => HirBinaryOp::Sub,
BinaryOp::Mul => HirBinaryOp::Mul,
BinaryOp::Div => HirBinaryOp::Div,
BinaryOp::Mod => HirBinaryOp::Mod,
BinaryOp::Eq => HirBinaryOp::Eq,
BinaryOp::Ne => HirBinaryOp::Ne,
BinaryOp::Lt => HirBinaryOp::Lt,
BinaryOp::Le => HirBinaryOp::Le,
BinaryOp::Gt => HirBinaryOp::Gt,
BinaryOp::Ge => HirBinaryOp::Ge,
BinaryOp::And => HirBinaryOp::And,
BinaryOp::Or => HirBinaryOp::Or,
BinaryOp::BitAnd => HirBinaryOp::And, // HIR-এ পৃথক নয়
BinaryOp::BitOr => HirBinaryOp::Or,
BinaryOp::BitXor => HirBinaryOp::Add, // প্লেসহোল্ডার
BinaryOp::Shl => HirBinaryOp::Add,
BinaryOp::Shr => HirBinaryOp::Add,
};
let ty = left.ty.clone();
Ok(HirExpr {
kind: HirExprKind::Binary(left, hir_op, right),
ty,
span: *span,
})
}
Expr::Call(func, args, span) => {
let func = Box::new(self.lower_expr(func)?);
let hir_args = args.iter().map(|a| self.lower_expr(a)).collect::<Result<Vec<_>, _>>()?;
let ty = func.ty.clone();
Ok(HirExpr {
kind: HirExprKind::Call(func, hir_args),
ty,
span: *span,
})
}
Expr::Block(block, span) => {
let hir_block = Box::new(self.lower_block(block)?);
Ok(HirExpr {
kind: HirExprKind::Block(hir_block),
ty: HirType::Unit, // ব্লকের শেষ এক্সপ্রেশনের টাইপ বের করা দরকার, আপাতত ইউনিট
span: *span,
})
}
Expr::Paren(e, _) => self.lower_expr(e), // পারেন উপেক্ষা
Expr::Error(span) => Ok(HirExpr {
kind: HirExprKind::Literal(HirLiteral::Unit),
ty: HirType::Error,
span: *span,
}),
_ => Err(format!("unsupported expression: {:?}", expr)),
}
}

// ----- প্যাটার্ন -----
fn lower_pat(&mut self, pat: &Pat) -> Result<HirPattern, String> {
match pat {
Pat::Wildcard(_) => Ok(HirPattern::Wildcard),
Pat::Ident(name, is_ref, is_mut, span) => {
// HIR-এ আমরা রেফ/মিউট ট্র্যাক করি না (সিম্পলিফাইড)
let ty = HirType::Error; // পরে টাইপ ইনফারেন্স থেকে আসবে
Ok(HirPattern::Variable(name.clone(), ty))
}
Pat::Tuple(pats, _) => {
let mut hir_pats = Vec::new();
for p in pats {
hir_pats.push(self.lower_pat(p)?);
}
Ok(HirPattern::Tuple(hir_pats))
}
Pat::Struct(fields, _) => {
let mut hir_fields = Vec::new();
for (fname, p) in fields {
hir_fields.push((fname.clone(), self.lower_pat(p)?));
}
Ok(HirPattern::Struct("".to_string(), hir_fields)) // নাম ঠিক করতে হবে
}
Pat::Error(_) => Ok(HirPattern::Wildcard),
_ => Err(format!("unsupported pattern: {:?}", pat)),
}
}

// ----- টাইপ -----
fn lower_ty(&self, ty: &Ty) -> HirType {
match ty {
Ty::Path(name, _) => {
match name.as_str() {
"int" => HirType::Int,
"float" => HirType::Float,
"string" => HirType::String,
"bool" => HirType::Bool,
"char" => HirType::Char,
"unit" | "()" => HirType::Unit,
_ => {
// টাইপ ম্যাপে দেখি
if let Some(t) = self.type_map.get(name) {
t.clone()
} else {
HirType::Generic(name.clone())
}
}
}
}
Ty::Tuple(types, _) => {
let inner = types.iter().map(|t| self.lower_ty(t)).collect();
HirType::Tuple(inner)
}
Ty::Array(elem, _) => HirType::Array(Box::new(self.lower_ty(elem))),
Ty::Ptr(elem, _) => HirType::Ptr(Box::new(self.lower_ty(elem))),
Ty::Ref(elem, is_mut, _) => HirType::Ptr(Box::new(self.lower_ty(elem))), // HIR-এ রেফ আলাদা নয়
Ty::Generic(name, _) => HirType::Generic(name.clone()),
Ty::Error(_) => HirType::Error,
}
}
}
