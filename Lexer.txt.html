```rust
//! # Source Span
//!
//! Represents a contiguous range of bytes in the source code.

use std::fmt;

#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct Span {
pub start: usize,
pub end: usize,
}

impl Span {
pub const fn new(start: usize, end: usize) -> Self {
Span { start, end }
}

pub fn merge(self, other: Self) -> Self {
Span {
start: self.start.min(other.start),
end: self.end.max(other.end),
}
}

pub const fn len(&self) -> usize {
self.end - self.start
}

pub const fn is_empty(&self) -> bool {
self.start == self.end
}

pub fn contains(&self, byte: usize) -> bool {
self.start <= byte && byte < self.end
}
}

impl Default for Span {
fn default() -> Self {
Span { start: 0, end: 0 }
}
}

impl fmt::Display for Span {
fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
write!(f, "{}..{}", self.start, self.end)
}
}
```

```rust
//! # Source Position
//!
//! Tracks line, column, and byte offset.

#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
pub struct Position {
pub line: usize,
pub col: usize,
pub byte: usize,
}

impl Position {
pub const fn new(line: usize, col: usize, byte: usize) -> Self {
Position { line, col, byte }
}

pub const fn start() -> Self {
Position::new(1, 1, 0)
}

pub fn advance(&mut self, c: char) {
if c == '\n' {
self.line += 1;
self.col = 1;
} else {
self.col += 1;
}
self.byte += c.len_utf8();
}

pub const fn line(&self) -> usize {
self.line
}
pub const fn col(&self) -> usize {
self.col
}
pub const fn byte(&self) -> usize {
self.byte
}
}
```

```rust
//! # Main Tokenizer Struct
//!
//! Defines the Tokenizer struct and the `next_token` entry point.

use crate::lexer::{
helpers::*, span::Span, position::Position, token::{Token, TokenKind},
};
use std::iter::Peekable;
use std::str::Chars;

pub struct Tokenizer<'src> {
pub(super) source: &'src str,
pub(super) chars: Peekable<Chars<'src>>,
pub(super) pos: Position,
}

impl<'src> Tokenizer<'src> {
pub fn new(source: &'src str) -> Self {
Tokenizer {
source,
chars: source.chars().peekable(),
pos: Position::start(),
}
}

pub(super) fn advance(&mut self) -> Option<char> {
let c = self.chars.next()?;
self.pos.advance(c);
Some(c)
}

pub(super) fn peek(&mut self) -> Option<char> {
self.chars.peek().copied()
}

pub(super) fn peek_nth(&mut self, n: usize) -> Option<char> {
let mut iter = self.chars.clone();
for _ in 0..n {
iter.next()?;
}
iter.peek().copied()
}

pub(super) fn current_byte(&self) -> usize {
self.pos.byte()
}

pub(super) fn make_span(&self, start: usize) -> Span {
Span::new(start, self.current_byte())
}

/// Fetches the next token from the source.
pub fn next_token(&mut self) -> Token {
self.skip_whitespace_and_comments();

let start = self.current_byte();
let c = match self.peek() {
Some(ch) => ch,
None => return Token::new(TokenKind::Eof, Span::new(start, start)),
};

let kind = if is_ident_start(c) {
// Handle 'b' and 'r' prefixes for byte/raw strings
match c {
'b' => {
let next1 = self.peek_nth(1);
let next2 = self.peek_nth(2);
if next1 == Some('"')
|| (next1 == Some('r') && (next2 == Some('"') || next2 == Some('#')))
{
self.lex_string_or_byte()
} else {
self.lex_identifier_or_keyword()
}
}
'r' => {
let next1 = self.peek_nth(1);
if next1 == Some('"') || next1 == Some('#') {
self.lex_string_or_byte()
} else {
self.lex_identifier_or_keyword()
}
}
_ => self.lex_identifier_or_keyword(),
}
} else if is_digit(c) {
self.lex_number()
} else {
match c {
'"' => self.lex_string(),
'\'' => self.lex_lifetime_or_char(),
'+' => self.lex_plus(),
'-' => self.lex_minus(),
'*' => self.lex_star(),
'/' => self.lex_slash(),
'%' => self.lex_percent(),
'^' => self.lex_xor(),
'&' => self.lex_and(),
'|' => self.lex_or(),
'!' => self.lex_not(),
'=' => self.lex_eq(),
'<' => self.lex_lt(),
'>' => self.lex_gt(),
'.' => self.lex_dot(),
':' => self.lex_colon(),
'(' => {
self.advance();
TokenKind::LParen
}
')' => {
self.advance();
TokenKind::RParen
}
'{' => {
self.advance();
TokenKind::LBrace
}
'}' => {
self.advance();
TokenKind::RBrace
}
'[' => {
self.advance();
TokenKind::LBracket
}
']' => {
self.advance();
TokenKind::RBracket
}
',' => {
self.advance();
TokenKind::Comma
}
';' => {
self.advance();
TokenKind::Semicolon
}
'#' => {
self.advance();
TokenKind::Hash
}
// ===== Hardware punctuators =====
'@' | '$' | '~' | '?' | '`' => self.lex_hw_punctuator(),
'_' => {
self.advance();
TokenKind::Underscore
}
_ => {
self.advance();
return Token::new(
TokenKind::Error(format!("Unexpected character '{}'", c)),
self.make_span(start),
);
}
}
};

Token::new(kind, self.make_span(start))
}
}
```

```rust
//! # Tokenizer Implementation (Part 1)
//!
//! Implements comment skipping, identifier/keyword lexing, and number lexing.

use super::tokenizer::Tokenizer;
use super::token::TokenKind;
use super::helpers::*;
use crate::lexer::tables::keywords::{KEYWORDS, BANGLA_KEYWORDS};
use crate::lexer::tables::hw_keywords::HW_KEYWORDS;

impl<'src> Tokenizer<'src> {
/// Skips whitespace and comments (line and block).
pub(super) fn skip_whitespace_and_comments(&mut self) {
while let Some(c) = self.peek() {
if is_whitespace(c) {
self.advance();
continue;
}
if c == '/' {
if let Some(next) = self.peek_nth(1) {
if next == '/' {
self.skip_line_comment();
continue;
} else if next == '*' {
self.skip_block_comment();
continue;
}
}
}
break;
}
}

/// Skips a single-line comment (`// ...`).
fn skip_line_comment(&mut self) {
self.advance(); // '/'
self.advance(); // '/'
while let Some(ch) = self.peek() {
if ch == '\n' {
break;
}
self.advance();
}
}

/// Skips a nested block comment (`/* ... */`).
fn skip_block_comment(&mut self) {
self.advance(); // '/'
self.advance(); // '*'
let mut depth = 1;
while depth > 0 {
if let Some(ch) = self.peek() {
self.advance();
if ch == '*' && self.peek() == Some('/') {
self.advance();
depth -= 1;
} else if ch == '/' && self.peek() == Some('*') {
self.advance();
depth += 1;
}
} else {
break;
}
}
}

/// Lexes an identifier or keyword (English, Bangla, or Hardware).
pub(super) fn lex_identifier_or_keyword(&mut self) -> TokenKind {
let mut ident = String::new();
while let Some(c) = self.peek() {
if !is_ident_continue(c) {
break;
}
ident.push(c);
self.advance();
}

// Check order: English -> Bangla -> Hardware
if let Some(&kind) = KEYWORDS.get(ident.as_str()) {
return kind;
}
if let Some(&kind) = BANGLA_KEYWORDS.get(ident.as_str()) {
return kind;
}
if let Some(&kind) = HW_KEYWORDS.get(ident.as_str()) {
return kind;
}
TokenKind::Ident(ident)
}

/// Lexes a number (integer, hex, binary, octal, float).
pub(super) fn lex_number(&mut self) -> TokenKind {
let mut num = String::new();
let mut is_float = false;

// Hex / Binary / Octal detection
if self.peek() == Some('0') {
let next = self.peek_nth(1);
if next == Some('x') || next == Some('X') {
num.push('0');
num.push('x');
self.advance();
self.advance();
while let Some(c) = self.peek() {
if !is_hex_digit(c) && c != '_' {
break;
}
num.push(c);
self.advance();
}
return TokenKind::IntLit(num);
}
if next == Some('b') || next == Some('B') {
num.push('0');
num.push('b');
self.advance();
self.advance();
while let Some(c) = self.peek() {
if !is_bin_digit(c) && c != '_' {
break;
}
num.push(c);
self.advance();
}
return TokenKind::IntLit(num);
}
if next == Some('o') || next == Some('O') {
num.push('0');
num.push('o');
self.advance();
self.advance();
while let Some(c) = self.peek() {
if !is_oct_digit(c) && c != '_' {
break;
}
num.push(c);
self.advance();
}
return TokenKind::IntLit(num);
}
}

// Decimal / Float
while let Some(c) = self.peek() {
if is_digit(c) {
num.push(c);
self.advance();
} else if c == '.' && !is_float && self.peek_nth(1).map_or(false, |n| is_digit(n)) {
is_float = true;
num.push(c);
self.advance();
} else if c == 'e' || c == 'E' {
is_float = true;
num.push(c);
self.advance();
if self.peek() == Some('+') || self.peek() == Some('-') {
num.push(self.peek().unwrap());
self.advance();
}
while let Some(d) = self.peek() {
if !is_digit(d) && d != '_' {
break;
}
num.push(d);
self.advance();
}
break;
} else if c == '_' {
self.advance();
continue;
} else {
break;
}
}

if is_float {
TokenKind::FloatLit(num)
} else {
TokenKind::IntLit(num)
}
}
}
```

```rust
//! # Tokenizer Implementation (Part 3)
//!
//! Implements lexing for strings, characters, and lifetimes.

use super::tokenizer::Tokenizer;
use super::token::TokenKind;
use super::helpers::*;
use crate::lexer::tables::escape_tables::decode_unicode_escape;

impl<'src> Tokenizer<'src> {
/// Lexes a lifetime ('ident) or character literal.
pub(super) fn lex_lifetime_or_char(&mut self) -> TokenKind {
self.advance(); // consume '\''

// Check for lifetime: 'ident
if let Some(c) = self.peek() {
if is_ident_start(c) && c != '\\' {
let mut lifetime = String::new();
while let Some(ch) = self.peek() {
if !is_ident_continue(ch) {
break;
}
lifetime.push(ch);
self.advance();
}
return TokenKind::Lifetime(lifetime);
}
}

// Character literal
let ch = match self.peek() {
Some('\\') => {
self.advance();
let esc = self.peek().unwrap_or('?');
self.advance();
match esc {
'n' => '\n',
't' => '\t',
'r' => '\r',
'0' => '\0',
'\\' => '\\',
'\'' => '\'',
'"' => '"',
_ => '?',
}
}
Some(c) => {
self.advance();
c
}
None => return TokenKind::Error("Unclosed character literal".to_string()),
};

if self.peek() == Some('\'') {
self.advance();
TokenKind::CharLit(ch)
} else {
TokenKind::Error("Expected closing quote".to_string())
}
}

/// Lexes a string or byte string (entry point).
pub(super) fn lex_string_or_byte(&mut self) -> TokenKind {
let is_byte = self.peek() == Some('b');
let is_raw = self.peek() == Some('r');

if is_byte {
self.advance(); // 'b'
if self.peek() == Some('"') {
return self.lex_byte_string();
} else if self.peek() == Some('r') {
self.advance(); // 'r'
return self.lex_raw_byte_string();
}
}

if is_raw {
self.advance(); // 'r'
return self.lex_raw_string();
}

// Normal string
self.lex_string()
}

/// Lexes a normal string (with escape sequences).
pub(super) fn lex_string(&mut self) -> TokenKind {
self.advance(); // consume '"'
let mut s = String::new();

while let Some(c) = self.peek() {
if c == '"' {
self.advance();
break;
}
if c == '\\' {
self.advance();
if let Some(esc) = self.peek() {
match esc {
'n' => {
s.push('\n');
self.advance();
}
't' => {
s.push('\t');
self.advance();
}
'r' => {
s.push('\r');
self.advance();
}
'0' => {
s.push('\0');
self.advance();
}
'\\' => {
s.push('\\');
self.advance();
}
'"' => {
s.push('"');
self.advance();
}
'x' => {
self.advance();
let mut hex = String::new();
for _ in 0..2 {
if let Some(d) = self.peek() {
if d.is_ascii_hexdigit() {
hex.push(d);
self.advance();
} else {
break;
}
}
}
if let Ok(v) = u8::from_str_radix(&hex, 16) {
s.push(v as char);
}
}
'u' => {
self.advance();
if self.peek() != Some('{') {
break;
}
self.advance();
let mut hex = String::new();
while let Some(c) = self.peek() {
if c == '}' {
break;
}
if !c.is_ascii_hexdigit() {
return TokenKind::Error("Invalid unicode escape".to_string());
}
hex.push(c);
self.advance();
}
if self.peek() != Some('}') {
return TokenKind::Error("Unclosed unicode escape".to_string());
}
self.advance();
if let Some(ch) = decode_unicode_escape(&hex) {
s.push(ch);
}
}
_ => {
s.push(esc);
self.advance();
}
}
}
} else {
s.push(c);
self.advance();
}
}
TokenKind::StringLit(s)
}
}
```

```rust
//! # Tokenizer Implementation (Part 4)
//!
//! Implements lexing for raw strings and byte strings.

use super::tokenizer::Tokenizer;
use super::token::TokenKind;

impl<'src> Tokenizer<'src> {
/// Lexes a raw string (r#"..."#).
pub(super) fn lex_raw_string(&mut self) -> TokenKind {
let hash_count = self.count_raw_hash_delimiters();
if self.peek() != Some('"') {
return TokenKind::Error("Expected '\"' after raw string prefix".to_string());
}
self.advance(); // consume '"'

let mut content = String::new();
while let Some(c) = self.peek() {
if c == '"' {
self.advance();
let mut close_ok = true;
for _ in 0..hash_count {
if self.peek() == Some('#') {
self.advance();
} else {
close_ok = false;
break;
}
}
if close_ok {
return TokenKind::RawString(content);
} else {
content.push('"');
for _ in 0..hash_count {
if let Some('#') = self.peek() {
self.advance();
content.push('#');
}
}
}
} else {
content.push(c);
self.advance();
}
}
TokenKind::Error("Unclosed raw string".to_string())
}

/// Lexes a raw byte string (br#"..."#).
pub(super) fn lex_raw_byte_string(&mut self) -> TokenKind {
let hash_count = self.count_raw_hash_delimiters();
if self.peek() != Some('"') {
return TokenKind::Error("Expected '\"' after raw byte string prefix".to_string());
}
self.advance(); // consume '"'

let mut content = Vec::new();
while let Some(c) = self.peek() {
if c == '"' {
self.advance();
let mut close_ok = true;
for _ in 0..hash_count {
if self.peek() == Some('#') {
self.advance();
} else {
close_ok = false;
break;
}
}
if close_ok {
return TokenKind::ByteStringLit(content);
} else {
content.push(b'"');
for _ in 0..hash_count {
if let Some('#') = self.peek() {
self.advance();
content.push(b'#');
}
}
}
} else {
content.push(c as u8);
self.advance();
}
}
TokenKind::Error("Unclosed raw byte string".to_string())
}

/// Lexes a byte string (b"...").
pub(super) fn lex_byte_string(&mut self) -> TokenKind {
self.advance(); // consume '"'
let mut bytes = Vec::new();

while let Some(c) = self.peek() {
if c == '"' {
self.advance();
break;
}
if c == '\\' {
self.advance();
if let Some(esc) = self.peek() {
let b = match esc {
'n' => b'\n',
't' => b'\t',
'r' => b'\r',
'0' => b'\0',
'\\' => b'\\',
'"' => b'"',
'x' => {
self.advance();
let mut hex = String::new();
for _ in 0..2 {
if let Some(d) = self.peek() {
if d.is_ascii_hexdigit() {
hex.push(d);
self.advance();
} else {
break;
}
}
}
u8::from_str_radix(&hex, 16).unwrap_or(b'?')
}
_ => esc as u8,
};
bytes.push(b);
self.advance();
}
} else {
bytes.push(c as u8);
self.advance();
}
}
TokenKind::ByteStringLit(bytes)
}

/// Counts raw hash delimiters (e.g., r#"..."# => 1).
pub(super) fn count_raw_hash_delimiters(&mut self) -> usize {
let mut count = 0;
while let Some('#') = self.peek() {
count += 1;
self.advance();
}
count
}
}
```
