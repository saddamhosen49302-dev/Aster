//! # Symbol Table for Name Resolution
//!
//! Defines symbols, scopes, and the main symbol table structure (merged from 086).

use std::collections::HashMap;
use crate::lexer::Span;
use crate::ast::Ty;

/// A symbol in the symbol table.
#[derive(Debug, Clone)]
pub struct Symbol {
pub name: String,
pub kind: SymbolKind,
pub span: Span,
pub ty: Option<Ty>,
pub is_mutable: bool,
}

impl Symbol {
pub fn new(name: String, kind: SymbolKind, span: Span) -> Self {
Symbol {
name,
kind,
span,
ty: None,
is_mutable: false,
}
}

pub fn with_type(mut self, ty: Ty) -> Self {
self.ty = Some(ty);
self
}
}

/// The kind of a symbol.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum SymbolKind {
Variable,
Function,
Struct,
Enum,
Trait,
Module,
Type,
GenericParam,
HwModule,
HwSignal,
HwParam,
}

/// A scope in the symbol table.
#[derive(Debug, Clone)]
pub struct Scope {
pub parent: Option<usize>,
pub symbols: HashMap<String, Symbol>,
}

impl Scope {
pub fn new(parent: Option<usize>) -> Self {
Scope {
parent,
symbols: HashMap::new(),
}
}

pub fn insert(&mut self, name: String, sym: Symbol) -> Option<Symbol> {
self.symbols.insert(name, sym)
}

pub fn lookup(&self, name: &str) -> Option<&Symbol> {
self.symbols.get(name)
}
}

/// A symbol table with nested scopes (এখন এখানেই সংযুক্ত)।
#[derive(Debug, Clone)]
pub struct SymbolTable {
scopes: Vec<Scope>,
current_scope: usize,
}

impl SymbolTable {
pub fn new() -> Self {
let root = Scope::new(None);
SymbolTable {
scopes: vec![root],
current_scope: 0,
}
}

pub fn enter_scope(&mut self) -> usize {
let idx = self.scopes.len();
let parent = self.current_scope;
self.scopes.push(Scope::new(Some(parent)));
self.current_scope = idx;
idx
}

pub fn leave_scope(&mut self) {
if let Some(parent) = self.scopes[self.current_scope].parent {
self.current_scope = parent;
}
}

pub fn insert(&mut self, name: String, sym: Symbol) -> Option<Symbol> {
self.scopes[self.current_scope].insert(name, sym)
}

pub fn lookup(&self, name: &str) -> Option<&Symbol> {
let mut idx = self.current_scope;
while let Some(scope) = self.scopes.get(idx) {
if let Some(sym) = scope.lookup(name) {
return Some(sym);
}
idx = match scope.parent {
Some(p) => p,
None => break,
};
}
None
}

pub fn lookup_current(&self, name: &str) -> Option<&Symbol> {
self.scopes[self.current_scope].lookup(name)
}

pub fn current_scope(&self) -> usize {
self.current_scope
}
}

//! # Name Resolver
//!
//! Walks the AST and populates the symbol table.

use crate::ast::*;
use crate::lexer::Span;
use super::symbol_table::{Symbol, SymbolKind, SymbolTable};
use super::diagnostics_integration::SemaDiagnostics;

pub struct Resolver {
symtab: SymbolTable,
}

impl Resolver {
pub fn new() -> Self {
Resolver {
symtab: SymbolTable::new(),
}
}

/// সিম্বল টেবিলে পাবলিক অ্যাক্সেস (টাইপ চেকার ব্যবহার করবে)
pub fn symtab(&self) -> &SymbolTable {
&self.symtab
}

pub fn resolve_file(&mut self, file: &File, diag: &mut SemaDiagnostics) -> Result<(), ()> {
for item in &file.items {
self.resolve_item(item, diag)?;
}
Ok(())
}

fn resolve_item(&mut self, item: &Item, diag: &mut SemaDiagnostics) -> Result<(), ()> {
match item {
Item::Fn(sig) => {
let sym = Symbol::new(sig.name.clone(), SymbolKind::Function, sig.span);
self.symtab.insert(sig.name.clone(), sym);
self.symtab.enter_scope();
for (pat, ty) in &sig.args {
if let Pat::Ident(name, _, _, span) = pat {
let var = Symbol::new(name.clone(), SymbolKind::Variable, *span)
.with_type(ty.clone());
self.symtab.insert(name.clone(), var);
}
}
self.resolve_block(&sig.body, diag)?;
self.symtab.leave_scope();
}
Item::Struct(name, fields, _, _, span) => {
let sym = Symbol::new(name.clone(), SymbolKind::Struct, *span);
self.symtab.insert(name.clone(), sym);
for f in fields {
let f_sym = Symbol::new(f.name.clone(), SymbolKind::Variable, f.span)
.with_type(f.ty.clone());
self.symtab.insert(f.name.clone(), f_sym);
}
}
Item::Enum(name, variants, _, _, span) => {
let sym = Symbol::new(name.clone(), SymbolKind::Enum, *span);
self.symtab.insert(name.clone(), sym);
}
Item::Impl(ty, _, _, _, body, _) => {
self.symtab.enter_scope();
self.resolve_block(body, diag)?;
self.symtab.leave_scope();
}
Item::Mod(_, items, _) => {
self.symtab.enter_scope();
for item in items {
self.resolve_item(item, diag)?;
}
self.symtab.leave_scope();
}
Item::Trait(_, _, _, _, _) => {}
_ => {}
}
Ok(())
}

fn resolve_block(&mut self, block: &Block, diag: &mut SemaDiagnostics) -> Result<(), ()> {
for stmt in &block.stmts {
self.resolve_stmt(stmt, diag)?;
}
Ok(())
}

fn resolve_stmt(&mut self, stmt: &Stmt, diag: &mut SemaDiagnostics) -> Result<(), ()> {
match stmt {
Stmt::Let(pat, _, expr, span) => {
if let Pat::Ident(name, _, _, pat_span) = pat {
if self.symtab.lookup_current(name).is_some() {
diag.error(format!("redefinition of `{}`", name), *pat_span);
} else {
let sym = Symbol::new(name.clone(), SymbolKind::Variable, *pat_span);
self.symtab.insert(name.clone(), sym);
}
}
self.resolve_expr(expr, diag)?;
}
Stmt::Expr(expr, _) | Stmt::Semi(expr, _) => {
self.resolve_expr(expr, diag)?;
}
Stmt::If(cond, then_b, else_b, _) => {
self.resolve_expr(cond, diag)?;
self.resolve_block(then_b, diag)?;
if let Some(e) = else_b {
self.resolve_stmt(e, diag)?;
}
}
Stmt::While(cond, body, _) => {
self.resolve_expr(cond, diag)?;
self.resolve_block(body, diag)?;
}
Stmt::For(pat, iter, body, _) => {
if let Pat::Ident(name, _, _, span) = pat {
let sym = Symbol::new(name.clone(), SymbolKind::Variable, *span);
self.symtab.insert(name.clone(), sym);
}
self.resolve_expr(iter, diag)?;
self.resolve_block(body, diag)?;
}
Stmt::Match(expr, arms, _) => {
self.resolve_expr(expr, diag)?;
for arm in arms {
self.resolve_pat(&arm.pat, diag)?;
if let Some(g) = &arm.guard {
self.resolve_expr(g, diag)?;
}
self.resolve_block(&arm.body, diag)?;
}
}
Stmt::Block(block, _) => {
self.symtab.enter_scope();
self.resolve_block(block, diag)?;
self.symtab.leave_scope();
}
_ => {}
}
Ok(())
}

fn resolve_expr(&mut self, expr: &Expr, diag: &mut SemaDiagnostics) -> Result<(), ()> {
match expr {
Expr::Variable(name, span) => {
if self.symtab.lookup(name).is_none() {
diag.error(format!("undefined variable `{}`", name), *span);
}
}
Expr::Binary(l, _, r, _) => {
self.resolve_expr(l, diag)?;
self.resolve_expr(r, diag)?;
}
Expr::Call(func, args, _) => {
self.resolve_expr(func, diag)?;
for arg in args {
self.resolve_expr(arg, diag)?;
}
}
Expr::Block(b, _) => {
self.symtab.enter_scope();
self.resolve_block(b, diag)?;
self.symtab.leave_scope();
}
_ => {}
}
Ok(())
}

fn resolve_pat(&mut self, pat: &Pat, diag: &mut SemaDiagnostics) -> Result<(), ()> {
match pat {
Pat::Ident(name, _, _, span) => {
if self.symtab.lookup_current(name).is_none() {
let sym = Symbol::new(name.clone(), SymbolKind::Variable, *span);
self.symtab.insert(name.clone(), sym);
}
}
Pat::Tuple(pats, _) => {
for p in pats {
self.resolve_pat(p, diag)?;
}
}
Pat::Struct(fields, _) => {
for (_, p) in fields {
self.resolve_pat(p, diag)?;
}
}
_ => {}
}
Ok(())
}
}

//! # Type Checking
//!
//! Performs type checking and basic type inference with symbol table access.

use crate::ast::*;
use crate::lexer::Span;
use super::symbol_table::SymbolTable;
use super::diagnostics_integration::SemaDiagnostics;

/// Semantic type representation.
#[derive(Debug, Clone, PartialEq)]
pub enum SemanticType {
Unit,
Bool,
Int,
Float,
String,
Char,
Ptr(Box<SemanticType>),
Array(Box<SemanticType>),
Tuple(Vec<SemanticType>),
Fn(Vec<SemanticType>, Box<SemanticType>),
Struct(String),
Enum(String),
Generic(String),
Error,
}

impl std::fmt::Display for SemanticType {
fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
match self {
SemanticType::Unit => write!(f, "()"),
SemanticType::Bool => write!(f, "bool"),
SemanticType::Int => write!(f, "int"),
SemanticType::Float => write!(f, "float"),
SemanticType::String => write!(f, "string"),
SemanticType::Char => write!(f, "char"),
SemanticType::Ptr(t) => write!(f, "*{}", t),
SemanticType::Array(t) => write!(f, "[{}]", t),
SemanticType::Tuple(ts) => write!(f, "({})", ts.iter().map(|t| t.to_string()).collect::<Vec<_>>().join(", ")),
SemanticType::Fn(args, ret) => write!(f, "fn({}) -> {}", args.iter().map(|t| t.to_string()).collect::<Vec<_>>().join(", "), ret),
SemanticType::Struct(n) => write!(f, "{}", n),
SemanticType::Enum(n) => write!(f, "{}", n),
SemanticType::Generic(n) => write!(f, "{}", n),
SemanticType::Error => write!(f, "<error>"),
}
}
}

pub struct TypeChecker;

impl TypeChecker {
pub fn new() -> Self {
TypeChecker
}

pub fn check_file(
&self,
file: &File,
symtab: &SymbolTable,
diag: &mut SemaDiagnostics,
) -> Result<(), ()> {
for item in &file.items {
self.check_item(item, symtab, diag)?;
}
Ok(())
}

fn check_item(
&self,
item: &Item,
symtab: &SymbolTable,
diag: &mut SemaDiagnostics,
) -> Result<(), ()> {
match item {
Item::Fn(sig) => {
self.check_block(&sig.body, symtab, diag)?;
}
Item::Impl(_, _, _, _, body, _) => {
self.check_block(body, symtab, diag)?;
}
_ => {}
}
Ok(())
}

fn check_block(
&self,
block: &Block,
symtab: &SymbolTable,
diag: &mut SemaDiagnostics,
) -> Result<(), ()> {
for stmt in &block.stmts {
self.check_stmt(stmt, symtab, diag)?;
}
Ok(())
}

fn check_stmt(
&self,
stmt: &Stmt,
symtab: &SymbolTable,
diag: &mut SemaDiagnostics,
) -> Result<(), ()> {
match stmt {
Stmt::Let(pat, ty_opt, expr, span) => {
let expr_ty = self.check_expr(expr, symtab, diag)?;
if let Some(ty) = ty_opt {
let expected = self.type_from_ast(ty);
if expr_ty != expected && expected != SemanticType::Error {
diag.error(
format!("type mismatch: expected {}, found {}", expected, expr_ty),
*span,
);
}
}
Ok(())
}
Stmt::Expr(expr, _) | Stmt::Semi(expr, _) => {
self.check_expr(expr, symtab, diag)?;
Ok(())
}
Stmt::Return(expr_opt, span) => {
if let Some(expr) = expr_opt {
self.check_expr(expr, symtab, diag)?;
}
Ok(())
}
Stmt::If(cond, then_b, else_b, span) => {
let cond_ty = self.check_expr(cond, symtab, diag)?;
if cond_ty != SemanticType::Bool {
diag.error("if condition must be bool".to_string(), *span);
}
self.check_block(then_b, symtab, diag)?;
if let Some(e) = else_b {
self.check_stmt(e, symtab, diag)?;
}
Ok(())
}
Stmt::While(cond, body, span) => {
let cond_ty = self.check_expr(cond, symtab, diag)?;
if cond_ty != SemanticType::Bool {
diag.error("while condition must be bool".to_string(), *span);
}
self.check_block(body, symtab, diag)?;
Ok(())
}
Stmt::Match(expr, arms, span) => {
self.check_expr(expr, symtab, diag)?;
for arm in arms {
if let Some(guard) = &arm.guard {
let guard_ty = self.check_expr(guard, symtab, diag)?;
if guard_ty != SemanticType::Bool {
diag.error("match guard must be bool".to_string(), arm.span);
}
}
self.check_block(&arm.body, symtab, diag)?;
}
Ok(())
}
_ => Ok(()),
}
}

fn check_expr(
&self,
expr: &Expr,
symtab: &SymbolTable,
diag: &mut SemaDiagnostics,
) -> Result<SemanticType, ()> {
match expr {
Expr::IntLit(_, _) => Ok(SemanticType::Int),
Expr::FloatLit(_, _) => Ok(SemanticType::Float),
Expr::StringLit(_, _) => Ok(SemanticType::String),
Expr::BoolLit(_, _) => Ok(SemanticType::Bool),
Expr::CharLit(_, _) => Ok(SemanticType::Char),
Expr::Variable(name, span) => {
// 🔥 সিম্বল টেবিল থেকে টাইপ নিয়ে আসা হচ্ছে
if let Some(sym) = symtab.lookup(name) {
if let Some(ty) = &sym.ty {
return Ok(self.type_from_ast(ty));
}
diag.error(format!("variable `{}` has no type annotation", name), *span);
return Ok(SemanticType::Error);
}
diag.error(format!("undefined variable `{}`", name), *span);
Ok(SemanticType::Error)
}
Expr::Binary(l, op, r, span) => {
let lt = self.check_expr(l, symtab, diag)?;
let rt = self.check_expr(r, symtab, diag)?;
match op {
BinaryOp::Add | BinaryOp::Sub | BinaryOp::Mul | BinaryOp::Div | BinaryOp::Mod => {
if lt == SemanticType::Int && rt == SemanticType::Int {
Ok(SemanticType::Int)
} else if lt == SemanticType::Float && rt == SemanticType::Float {
Ok(SemanticType::Float)
} else {
diag.error(format!("type mismatch: {:?} vs {:?}", lt, rt), *span);
Err(())
}
}
BinaryOp::Eq | BinaryOp::Ne | BinaryOp::Lt | BinaryOp::Le | BinaryOp::Gt | BinaryOp::Ge => {
if lt == rt {
Ok(SemanticType::Bool)
} else {
diag.error(format!("comparison type mismatch: {:?} vs {:?}", lt, rt), *span);
Err(())
}
}
BinaryOp::And | BinaryOp::Or => {
if lt == SemanticType::Bool && rt == SemanticType::Bool {
Ok(SemanticType::Bool)
} else {
diag.error("logical operator requires bool".to_string(), *span);
Err(())
}
}
_ => {
diag.error("unsupported binary operator".to_string(), *span);
Err(())
}
}
}
Expr::Unary(op, e, span) => {
let ty = self.check_expr(e, symtab, diag)?;
match op {
UnaryOp::Neg => {
if ty == SemanticType::Int || ty == SemanticType::Float {
Ok(ty)
} else {
diag.error("negation requires int or float".to_string(), *span);
Err(())
}
}
UnaryOp::Not => {
if ty == SemanticType::Bool {
Ok(SemanticType::Bool)
} else {
diag.error("logical not requires bool".to_string(), *span);
Err(())
}
}
UnaryOp::Deref => {
match ty {
SemanticType::Ptr(inner) => Ok(*inner),
_ => {
diag.error("deref requires pointer".to_string(), *span);
Err(())
}
}
}
}
}
Expr::Block(block, _) => {
self.check_block(block, symtab, diag)?;
Ok(SemanticType::Unit)
}
Expr::Paren(e, _) => self.check_expr(e, symtab, diag),
_ => Ok(SemanticType::Error),
}
}

fn type_from_ast(&self, ty: &Ty) -> SemanticType {
match ty {
Ty::Path(name, _) => match name.as_str() {
"int" => SemanticType::Int,
"float" => SemanticType::Float,
"string" => SemanticType::String,
"bool" => SemanticType::Bool,
"char" => SemanticType::Char,
"unit" | "()" => SemanticType::Unit,
_ => SemanticType::Generic(name.clone()),
},
Ty::Tuple(types, _) => {
let inner: Vec<SemanticType> = types.iter().map(|t| self.type_from_ast(t)).collect();
SemanticType::Tuple(inner)
}
Ty::Array(elem, _) => SemanticType::Array(Box::new(self.type_from_ast(elem))),

}
}

//!
//! Performs type checking and basic type inference with symbol table access.

use crate::lexer::Span;
use super::symbol_table::SymbolTable;
use super::diagnostics_integration::SemaDiagnostics;
use std::collections::HashMap;

/// Semantic type representation.
#[derive(Debug, Clone, PartialEq)]
pub enum SemanticType {
Unit,
Bool,
Int,
Float,
String,
Char,
Ptr(Box<SemanticType>),
Array(Box<SemanticType>),
Tuple(Vec<SemanticType>),
Fn(Vec<SemanticType>, Box<SemanticType>),

}

fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

SemanticType::Int => write!(f, "int"),
SemanticType::Float => write!(f, "float"),
SemanticType::String => write!(f, "string"),
SemanticType::Char => write!(f, "char"),

Expr::FloatLit(_, _) => Ok(SemanticType::Float),
Expr::StringLit(_, _) => Ok(SemanticType::String),
Expr::BoolLit(_, _) => Ok(SemanticType::Bool),
Expr::CharLit(_, _) => Ok(SemanticType::Char),

}
} else {
diag.error(format!("unknown function `{}`", name), *span);
Ok(SemanticType::Error)
}
} else {
diag.error("call expression must be a function name".to_string(), *span);
Ok(SemanticType::Error)
}
}

let rt = self.check_expr(r, symtab, diag)?;

BinaryOp::Add | BinaryOp::Sub | BinaryOp::Mul | BinaryOp::Div | BinaryOp::Mod => {
if lt == SemanticType::Int && rt == SemanticType::Int {
Ok(SemanticType::Int)
} else if lt == SemanticType::Float && rt == SemanticType::Float {
Ok(SemanticType::Float)
} else {

Err(())
}

}

}
}
_ => {
diag.error("unsupported binary operator".to_string(), *span);
Err(())
}
}
}
Expr::Unary(op, e, span) => {
let ty = self.check_expr(e, symtab, diag)?;
match op {
UnaryOp::Neg => {

} else {
diag.error("negation requires int or float".to_string(), *span);
Err(())
}
}
UnaryOp::Not => {

Ok(SemanticType::Bool)
} else {
diag.error("logical not requires bool".to_string(), *span);
Err(())
}
}
UnaryOp::Deref => {
match ty {
SemanticType::Ptr(inner) => Ok(*inner),
_ => {
diag.error("deref requires pointer".to_string(), *span);
Err(())
}
}
}
}
}

// to the expression's type; in Rust, Semi yields Unit.
// But for block value, only Expr (without semicolon) matters.
// For simplicity, we treat both as yielding the expression type.
last_ty = ty;
}

self.check_expr(expr, symtab, diag)?;
last_ty = SemanticType::Unit; // Let statement doesn't yield value
}
// Other statements (if, while, etc.) may yield Unit or nothing
_ => {
// Recursively check and assume Unit
self.check_stmt(stmt, symtab, diag)?;
last_ty = SemanticType::Unit;
}
}
}
Ok(last_ty)
}
Expr::Paren(e, _) => self.check_expr(e, symtab, diag),
_ => Ok(SemanticType::Error),
}
}

}

}

}

}

}
_ => {}
}

.iter()
.filter_map(|(k, v)| if *v == self.current_depth + 1 { Some(k.clone()) } else { None })

}
