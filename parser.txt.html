self.active

while !parser.at(TokenKind::Semicolon)
&& !parser.at(TokenKind::RBrace)
&& !parser.at(TokenKind::Eof)

parser.bump();
}
if parser.at(TokenKind::Semicolon) {
parser.bump();
}
self.reset();
}

pub fn parse_expr_precedence(&mut self, min_prec: u8) -> Expr {
let mut left = self.parse_primary();

loop {
let op = match self.current_kind() {
TokenKind::Plus => BinaryOp::Add,
TokenKind::Minus => BinaryOp::Sub,
TokenKind::Star => BinaryOp::Mul,
TokenKind::Slash => BinaryOp::Div,
TokenKind::Percent => BinaryOp::Mod,
TokenKind::Eq => BinaryOp::Eq,
TokenKind::Neq => BinaryOp::Ne,
TokenKind::Lt => BinaryOp::Lt,
TokenKind::Le => BinaryOp::Le,
TokenKind::Gt => BinaryOp::Gt,
TokenKind::Ge => BinaryOp::Ge,
TokenKind::AndAnd => BinaryOp::And,
TokenKind::OrOr => BinaryOp::Or,
TokenKind::And => BinaryOp::BitAnd,
TokenKind::Or => BinaryOp::BitOr,
TokenKind::Xor => BinaryOp::BitXor,
TokenKind::LeftShift => BinaryOp::Shl,
TokenKind::RightShift => BinaryOp::Shr,
_ => break,
};

let prec = precedence(op);
if prec < min_prec {
break;
}

self.bump();
let rhs = self.parse_expr_precedence(prec + 1);
let span = left.span().merge(rhs.span());
left = Expr::Binary(Box::new(left), op, Box::new(rhs), span);
}

left
}

Stmt::Return(Some(Box::new(e)), span)

let mut clauses = Vec::new();
while !self.at(TokenKind::LBrace) && !self.at(TokenKind::Eof) {
let name = self.expect_ident();
self.expect(TokenKind::Colon);
let bound = self.expect_ident();
clauses.push((name, bound));
if self.at(TokenKind::Comma) {
self.bump();
}
}
clauses
}

self.expect(TokenKind::LParen);

self.expect(TokenKind::Colon);

if self.at(TokenKind::RParen) { break; }

use super::parser::Parser;
use crate::ast::Ty;
use crate::lexer::TokenKind;

impl<'src> Parser<'src> {
pub fn parse_ty(&mut self) -> Ty {
let span = self.current_span();

match self.current_kind() {
TokenKind::Ident(_) => {
let name = self.current_ident().unwrap();
self.bump();
// Check for generics like Vec<T>
let generics = if self.at(TokenKind::Lt) {
let mut params = Vec::new();
self.bump();
while !self.at(TokenKind::Gt) && !self.at(TokenKind::Eof) {
params.push(self.parse_ty());
if self.at(TokenKind::Comma) { self.bump(); }
}
self.expect(TokenKind::Gt);
Some(params)
} else { None };

// Heuristic: single uppercase = generic param, otherwise path
if name.chars().count() == 1 && name.chars().next().unwrap().is_uppercase() && generics.is_none() {
Ty::Generic(name, span)
} else {
Ty::Path(name, span)
}
}
TokenKind::LParen => {
self.bump();
let mut inner = Vec::new();
while !self.at(TokenKind::RParen) && !self.at(TokenKind::Eof) {
inner.push(self.parse_ty());
if self.at(TokenKind::Comma) { self.bump(); }
}
self.expect(TokenKind::RParen);
let end = self.current_span().end;
Ty::Tuple(inner, Span::new(span.start, end))
}
TokenKind::LBracket => {
self.bump();
let elem = self.parse_ty();
self.expect(TokenKind::RBracket);
let end = self.current_span().end;
Ty::Array(Box::new(elem), Span::new(span.start, end))
}
TokenKind::Star => {
self.bump();
let inner = self.parse_ty();
let end = self.current_span().end;
Ty::Ptr(Box::new(inner), Span::new(span.start, end))
}
TokenKind::And => {
self.bump();
let is_mut = self.at(TokenKind::Mut);
if is_mut { self.bump(); }
let inner = self.parse_ty();
let end = self.current_span().end;
Ty::Ref(Box::new(inner), is_mut, Span::new(span.start, end))
}
_ => {
self.error("Expected type");
self.bump();
Ty::Error(span)
}
}
}

pub fn parse_path(&mut self) -> String {
let mut path = self.expect_ident();
while self.at(TokenKind::PathSep) {
self.bump();
path.push_str("::");
path.push_str(&self.expect_ident());
}
path
}
