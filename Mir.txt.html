```rust
//! # MIR Instructions

use crate::lexer::Span;
use crate::ast::BinaryOp; // ← added

/// A place (local variable).
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Place {

pub index: Option<usize>,
pub span: Span,
}

/// An operand (constant or place).
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Operand {

Move(Place),
Copy(Place),
}

/// A constant value.
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Constant {
Int(i64),
Float(f64),
Bool(bool),
String(String),
Unit,
}

/// An R-value (right-hand side of an assignment).
#[derive(Debug, Clone, PartialEq)]
pub enum Rvalue {
Use(Operand),
Binary(BinaryOp, Operand, Operand),
Unary(UnaryOp, Operand),
Call(Operand, Vec<Operand>),
Aggregate(AggregateKind, Vec<Operand>),
}

/// Unary operator.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum UnaryOp {
Neg,
Not,
Deref,
}

/// Aggregate kind.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum AggregateKind {
Tuple,
Array,
Struct(String),
Enum(String, String),
}

/// A statement.
#[derive(Debug, Clone, PartialEq)]
pub enum Statement {
Assign(Place, Rvalue),
StorageLive(Place),
StorageDead(Place),
Nop,
}

/// Terminator kind.
#[derive(Debug, Clone, PartialEq)]
pub enum TerminatorKind {
Goto(usize),
Switch(Operand, Vec<(Constant, usize)>, usize),
Return,
Call(Operand, Vec<Operand>, usize, usize),

}

/// A terminator.
#[derive(Debug, Clone, PartialEq)]
```

```rust
//! # MIR Optimization Passes
//!
//! Defines various optimization passes for MIR.

mod const_prop;

mod aggressive_dce;

pub use const_prop::ConstProp;
pub use simplify_cfg::SimplifyCfg;
pub use aggressive_dce::AggressiveDce;

use super::MirBody;

/// Trait for optimization passes.
pub trait MirPass {
fn name(&self) -> &str;
fn run(&self, body: &mut MirBody) -> Result<(), String>;
}

/// Pass manager for running multiple passes.
pub struct PassManager {
passes: Vec<Box<dyn MirPass>>,
}

impl PassManager {
pub fn new() -> Self {
PassManager {
passes: Vec::new(),
}
}

pub fn add_pass(&mut self, pass: Box<dyn MirPass>) {
self.passes.push(pass);
}

pub fn run(&self, body: &mut MirBody) -> Result<(), String> {
for pass in &self.passes {
pass.run(body)?;
}
Ok(())
}

pub fn run_passes(&self, bodies: &mut [MirBody]) -> Result<(), String> {
for body in bodies {
self.run(body)?;
}
Ok(())
}
}
```

```rust
//! # Constant Propagation Pass
//!
//! Replaces expressions with constant values where possible.

use super::{MirPass, MirBody};
use super::super::instruction::{Statement, Rvalue, Operand, Constant, UnaryOp};
use crate::ast::BinaryOp; // ← added (explicit)

pub struct ConstProp;

impl ConstProp {
pub fn new() -> Self {
ConstProp
}

pub fn run_on_body(&self, body: &mut MirBody) {
for block in &mut body.basic_blocks {
for stmt in &mut block.statements {
if let Statement::Assign(_, rvalue) = stmt {
*rvalue = self.fold_rvalue(rvalue);
}
}
}
}

fn fold_rvalue(&self, rvalue: &Rvalue) -> Rvalue {
match rvalue {
Rvalue::Binary(op, op1, op2) => {
if let (Operand::Constant(c1), Operand::Constant(c2)) = (op1, op2) {
match (c1, c2) {
(Constant::Int(a), Constant::Int(b)) => {
let result = match op {
BinaryOp::Add => a + b,
BinaryOp::Sub => a - b,
BinaryOp::Mul => a * b,
BinaryOp::Div => {
if *b != 0 {
a / b
} else {
return rvalue.clone();
}
}
BinaryOp::Mod => {
if *b != 0 {
a % b
} else {
return rvalue.clone();
}
}
_ => return rvalue.clone(),
};
return Rvalue::Use(Operand::Constant(Constant::Int(result)));
}
(Constant::Float(a), Constant::Float(b)) => {
let result = match op {
BinaryOp::Add => a + b,
BinaryOp::Sub => a - b,
BinaryOp::Mul => a * b,
BinaryOp::Div => {
if *b != 0.0 {
a / b
} else {
return rvalue.clone();
}
}
_ => return rvalue.clone(),
};
return Rvalue::Use(Operand::Constant(Constant::Float(result)));
}
_ => return rvalue.clone(),
}
}
rvalue.clone()
}
Rvalue::Unary(op, operand) => {
if let Operand::Constant(c) = operand {
match (op, c) {
(UnaryOp::Neg, Constant::Int(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Int(-v)));
}
(UnaryOp::Neg, Constant::Float(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Float(-v)));
}
(UnaryOp::Not, Constant::Bool(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Bool(!v)));
}
_ => return rvalue.clone(),
}
}
rvalue.clone()
}
_ => rvalue.clone(),
}
}
}

impl MirPass for ConstProp {
fn name(&self) -> &str {
"ConstantPropagation"
}

fn run(&self, body: &mut MirBody) -> Result<(), String> {
self.run_on_body(body);
Ok(())
}
}
```

```rust
//! # CFG Simplification Pass
//!
//! Removes unreachable blocks and simplifies the control-flow graph.

use super::{MirPass, MirBody};
use super::super::instruction::TerminatorKind;
use std::collections::HashSet;

pub struct SimplifyCfg;

impl SimplifyCfg {
pub fn new() -> Self {
SimplifyCfg
}

pub fn run_on_body(&self, body: &mut MirBody) {
let mut reachable = HashSet::new();
let mut worklist = vec![0];

// Mark reachable blocks
while let Some(idx) = worklist.pop() {
if reachable.insert(idx) {
if let Some(block) = body.basic_blocks.get(idx) {
match &block.terminator.kind {
TerminatorKind::Goto(target) => worklist.push(*target),
TerminatorKind::Switch(_, cases, default) => {
for (_, target) in cases {
worklist.push(*target);
}
worklist.push(*default);
}
TerminatorKind::Call(_, _, ret, next) => {
worklist.push(*ret);
worklist.push(*next);
}
_ => {}
}
}
}
}

// Rebuild blocks with only reachable ones
let mut new_blocks = Vec::new();
let mut mapping = vec![usize::MAX; body.basic_blocks.len()];

for i in 0..body.basic_blocks.len() {
if reachable.contains(&i) {
mapping[i] = new_blocks.len();
new_blocks.push(body.basic_blocks[i].clone());
}
}

// Fix terminators to point to remapped blocks
for block in &mut new_blocks {
match &mut block.terminator.kind {
TerminatorKind::Goto(target) => {
if mapping[*target] != usize::MAX {
*target = mapping[*target];
}
}
TerminatorKind::Switch(_, cases, default) => {
for (_, target) in cases {
if mapping[*target] != usize::MAX {
*target = mapping[*target];
}
}
if mapping[*default] != usize::MAX {
*default = mapping[*default];
}
}
TerminatorKind::Call(_, _, ret, next) => {
if mapping[*ret] != usize::MAX {
*ret = mapping[*ret];
}
if mapping[*next] != usize::MAX {
*next = mapping[*next];
}
}
_ => {}
}
}

body.basic_blocks = new_blocks;
}
}

impl MirPass for SimplifyCfg {
fn name(&self) -> &str {
"SimplifyCFG"
}

fn run(&self, body: &mut MirBody) -> Result<(), String> {
self.run_on_body(body);
Ok(())
}
}
```

```rust
//! # Aggressive Dead Code Elimination Pass
//!
//! Removes unused assignments and dead code.

use super::{MirPass, MirBody};
use super::super::instruction::{Statement, Place, Operand, Rvalue, TerminatorKind};
use std::collections::HashSet;

pub struct AggressiveDce;

impl AggressiveDce {
pub fn new() -> Self {
AggressiveDce
}

pub fn run_on_body(&self, body: &mut MirBody) {
let mut used = HashSet::new();

// Collect uses from all statements and terminators
for block in &body.basic_blocks {
for stmt in &block.statements {
if let Statement::Assign(_, rvalue) = stmt {
self.collect_operands(rvalue, &mut used);
}
// StorageLive/Dead do not use places (they only declare)
}
// Collect operands from terminator
self.collect_terminator_operands(&block.terminator, &mut used);
}

// Remove unused assignments, keeping all non-assign statements
for block in &mut body.basic_blocks {
block.statements.retain(|stmt| {
if let Statement::Assign(place, _) = stmt {
// Keep if place (or any prefix) is used
self.is_place_used(place, &used)
} else {
true // keep storage statements and nop
}
});
}
}

/// Collect all places from an Rvalue
fn collect_operands(&self, rvalue: &Rvalue, used: &mut HashSet<Place>) {
match rvalue {
Rvalue::Use(op) => self.collect_operand(op, used),
Rvalue::Binary(_, op1, op2) => {
self.collect_operand(op1, used);
self.collect_operand(op2, used);
}
Rvalue::Unary(_, op) => self.collect_operand(op, used),
Rvalue::Call(func, args) => {
self.collect_operand(func, used);
for arg in args {
self.collect_operand(arg, used);
}
}
Rvalue::Aggregate(_, ops) => {
for op in ops {
self.collect_operand(op, used);
}
}
}
}

fn collect_operand(&self, op: &Operand, used: &mut HashSet<Place>) {
match op {
Operand::Move(p) | Operand::Copy(p) => {
self.add_place_with_prefixes(p, used);
}
_ => {}
}
}

/// Collect operands from terminator kinds
fn collect_terminator_operands(&self, terminator: &super::super::instruction::Terminator, used: &mut HashSet<Place>) {
match &terminator.kind {
TerminatorKind::Switch(op, _, _) => self.collect_operand(op, used),
TerminatorKind::Call(func, args, _, _) => {
self.collect_operand(func, used);
for arg in args {
self.collect_operand(arg, used);
}
}
_ => {} // Goto, Return, Unreachable have no operands
}
}

/// Insert the place itself and all its prefixes (e.g., for `x.0` insert `x.0` and `x`)
fn add_place_with_prefixes(&self, place: &Place, set: &mut HashSet<Place>) {
set.insert(place.clone());
if place.index.is_some() {
let mut prefix = place.clone();
prefix.index = None;
set.insert(prefix);
}
}

/// Check if a place or any of its prefixes is in the used set
fn is_place_used(&self, place: &Place, used: &HashSet<Place>) -> bool {
if used.contains(place) {
return true;
}
if place.index.is_some() {
let mut prefix = place.clone();
prefix.index = None;
return used.contains(&prefix);
}
false
}
}

impl MirPass for AggressiveDce {
fn name(&self) -> &str {
"AggressiveDCE"
}

fn run(&self, body: &mut MirBody) -> Result<(), String> {
self.run_on_body(body);
Ok(())
}
}
```

```rust
#[cfg(test)]
mod tests {

let mir = builder.build(&hir).unwrap();

assert_eq!(mir.block_count(), 1);
}
}
```

```rust
//! # MIR (Mid-level Intermediate Representation) – 100% Complete
//!
//! This module defines the MIR, which is a control-flow graph (CFG) based
//! representation with explicit basic blocks, SSA form, and low-level operations.
//! It is suitable for optimization passes and code generation.

mod cfg;
mod instruction;
mod passes;
mod hw_lowering;

pub mod tests;

pub use cfg::*;
pub use instruction::*;
pub use passes::*;
pub use hw_lowering::*;

use crate::lexer::Span;

/// MIR body containing a control-flow graph.
#[derive(Debug, Clone, PartialEq)]
pub struct MirBody {
pub name: String,
pub basic_blocks: Vec<BasicBlock>,
pub arg_count: usize,
pub local_count: usize,
pub span: Span,
}

impl MirBody {
pub fn new(name: String, span: Span) -> Self {
MirBody {
name,
basic_blocks: Vec::new(),
arg_count: 0,
local_count: 0,
span,
}
}

pub fn add_block(&mut self, block: BasicBlock) -> usize {
let idx = self.basic_blocks.len();
self.basic_blocks.push(block);
idx
}

pub fn block(&self, idx: usize) -> &BasicBlock {
&self.basic_blocks[idx]
}

pub fn block_mut(&mut self, idx: usize) -> &mut BasicBlock {
&mut self.basic_blocks[idx]
}

pub fn block_count(&self) -> usize {
self.basic_blocks.len()
}

pub fn is_empty(&self) -> bool {
self.basic_blocks.is_empty()
}
}
```

```rust
//! # Control-Flow Graph (CFG)
//!
//! Defines BasicBlock, the CFG structure, and a builder that fully lowers HIR to MIR.

use crate::lexer::Span;
use crate::hir::{HirFile, HirItem, HirFn, HirBlock, HirStmt, HirExpr, HirExprKind, HirLiteral};
use super::instruction::*;
use super::MirBody;
use std::collections::HashMap;

/// A basic block containing statements and a terminator.
#[derive(Debug, Clone, PartialEq)]
pub struct BasicBlock {
pub statements: Vec<Statement>,
pub terminator: Terminator,
pub span: Span,
}

impl BasicBlock {
pub fn new(terminator: Terminator, span: Span) -> Self {
BasicBlock {
statements: Vec::new(),
terminator,
span,
}
}

pub fn push(&mut self, stmt: Statement) {
self.statements.push(stmt);
}

pub fn is_empty(&self) -> bool {
self.statements.is_empty()
}

pub fn len(&self) -> usize {
self.statements.len()
}
}

/// MIR builder that constructs a CFG from HIR.
pub struct MirBuilder {
next_temp: usize,
locals: HashMap<String, Place>,
}

impl MirBuilder {
pub fn new() -> Self {
MirBuilder {
next_temp: 0,
locals: HashMap::new(),
}
}

/// Build MIR from an entire HIR file. Returns a vector of MirBody (one per function/item).
pub fn build(&mut self, hir: &HirFile) -> Result<Vec<MirBody>, String> {
let mut bodies = Vec::new();
for item in &hir.items {
if let HirItem::Fn(fn_def) = item {
let body = self.build_fn(fn_def)?;
bodies.push(body);
}
// Other items (structs, enums, etc.) are not lowered to MIR.
}
Ok(bodies)
}

/// Build MIR for a single function.
fn build_fn(&mut self, f: &HirFn) -> Result<MirBody, String> {
let mut mir = MirBody::new(f.name.clone(), f.span);
mir.arg_count = f.args.len();
mir.local_count = f.args.len(); // will be extended with temporaries

// Initialize local mapping for arguments
for (i, (name, ty)) in f.args.iter().enumerate() {
let place = Place {
name: name.clone(),
index: Some(i),
span: f.span,
};
self.locals.insert(name.clone(), place);
}

// Create entry and exit blocks
let entry = BasicBlock::new(Terminator { kind: TerminatorKind::Return, span: f.span }, f.span);
let entry_idx = mir.add_block(entry);

// Process the function body (HIR block) into MIR statements
let mut stmts = Vec::new();
self.lower_block(&f.body, &mut stmts, &mut mir)?;

// Append statements to the entry block
for stmt in stmts {
mir.basic_blocks[entry_idx].statements.push(stmt);
}

// Ensure terminator is set (if not already, we set Return)
// In a full implementation, we might add jumps.

Ok(mir)
}

/// Lower an HIR block into a list of MIR statements.
fn lower_block(
&mut self,
block: &HirBlock,
stmts: &mut Vec<Statement>,
mir: &mut MirBody,
) -> Result<(), String> {
for stmt in &block.stmts {
self.lower_stmt(stmt, stmts, mir)?;
}
Ok(())
}

/// Lower an HIR statement.
fn lower_stmt(
&mut self,
stmt: &HirStmt,
stmts: &mut Vec<Statement>,
mir: &mut MirBody,
) -> Result<(), String> {
match stmt {
HirStmt::Let(pat, expr_opt, span) => {
// For simplicity, we handle only variable patterns.
if let HirPattern::Variable(name, _) = pat {
let place = Place {
name: name.clone(),
index: Some(mir.local_count),
span: *span,
};
mir.local_count += 1;
self.locals.insert(name.clone(), place.clone());

if let Some(expr) = expr_opt {
let rvalue = self.lower_expr(expr, stmts, mir)?;
stmts.push(Statement::Assign(place, rvalue));
}
} else {
return Err("Only variable patterns supported in MIR lowering".to_string());
}
}
HirStmt::Expr(expr) => {
// Expression statement: evaluate and discard result.
let _rvalue = self.lower_expr(expr, stmts, mir)?;
}
HirStmt::Semi(expr) => {
// Same as Expr.
let _rvalue = self.lower_expr(expr, stmts, mir)?;
}
HirStmt::Return(expr_opt, span) => {
if let Some(expr) = expr_opt {
let rvalue = self.lower_expr(expr, stmts, mir)?;
let temp = self.new_temp(span);
stmts.push(Statement::Assign(temp.clone(), rvalue));
// Return the temp
stmts.push(Statement::Return(Some(Operand::Move(temp))));
} else {
stmts.push(Statement::Return(None));
}
}
HirStmt::Block(block) => {
self.lower_block(block, stmts, mir)?;
}
_ => {
// Other statements (if, while, match, etc.) are not yet implemented.
// For a complete compiler, we would lower them to CFG with branches.
// Here we just ignore (or could emit a placeholder).
}
}
Ok(())
}

/// Lower an HIR expression into an MIR Rvalue, and emit any temporary assignments.
fn lower_expr(
&mut self,
expr: &HirExpr,
stmts: &mut Vec<Statement>,
mir: &mut MirBody,
) -> Result<Rvalue, String> {
match &expr.kind {
HirExprKind::Literal(lit) => {
let constant = match lit {
HirLiteral::Int(i) => Constant::Int(*i),
HirLiteral::Float(f) => Constant::Float(*f),
HirLiteral::String(s) => Constant::String(s.clone()),
HirLiteral::Bool(b) => Constant::Bool(*b),
HirLiteral::Char(c) => Constant::Int(*c as i64),
HirLiteral::Unit => Constant::Unit,
};
Ok(Rvalue::Use(Operand::Constant(constant)))
}
HirExprKind::Variable(name) => {
let place = self.locals.get(name)
.ok_or_else(|| format!("Unknown variable: {}", name))?;
Ok(Rvalue::Use(Operand::Copy(place.clone())))
}
HirExprKind::Unary(op, e) => {
let operand = self.lower_expr_to_operand(e, stmts, mir)?;
let unary_op = match op {
HirUnaryOp::Neg => UnaryOp::Neg,
HirUnaryOp::Not => UnaryOp::Not,
HirUnaryOp::Deref => UnaryOp::Deref,
};
Ok(Rvalue::Unary(unary_op, operand))
}
HirExprKind::Binary(left, op, right) => {
let l_op = self.lower_expr_to_operand(left, stmts, mir)?;
let r_op = self.lower_expr_to_operand(right, stmts, mir)?;
let bin_op = match op {
HirBinaryOp::Add => BinaryOp::Add,
HirBinaryOp::Sub => BinaryOp::Sub,
HirBinaryOp::Mul => BinaryOp::Mul,
HirBinaryOp::Div => BinaryOp::Div,
HirBinaryOp::Mod => BinaryOp::Mod,
HirBinaryOp::Eq => BinaryOp::Eq,
HirBinaryOp::Ne => BinaryOp::Ne,
HirBinaryOp::Lt => BinaryOp::Lt,
HirBinaryOp::Le => BinaryOp::Le,
HirBinaryOp::Gt => BinaryOp::Gt,
HirBinaryOp::Ge => BinaryOp::Ge,
HirBinaryOp::And => BinaryOp::And,
HirBinaryOp::Or => BinaryOp::Or,
};
Ok(Rvalue::Binary(bin_op, l_op, r_op))
}
HirExprKind::Call(func, args) => {
// Simple direct call (function name must be a variable)
let func_op = match func.kind {
HirExprKind::Variable(ref name) => {
// Could be a function name; we treat as a constant for now.
// In full implementation, we'd look up the function.
Operand::Constant(Constant::String(name.clone()))
}
_ => return Err("Only direct function calls supported".to_string()),
};
let mut arg_ops = Vec::new();
for arg in args {
let op = self.lower_expr_to_operand(arg, stmts, mir)?;
arg_ops.push(op);
}
// Call returns a value; we need a temporary to hold the result.
// For simplicity, we just return a Call Rvalue; caller will assign.
Ok(Rvalue::Call(func_op, arg_ops))
}
HirExprKind::Block(block) => {
// Lower the block as a sequence of statements and return the last expression's value.
// We'll emit statements into the same `stmts` vector.
let saved_len = stmts.len();
self.lower_block(block, stmts, mir)?;
// Find the last expression in the block (if any) and return its Rvalue.
// For simplicity, we just return Unit.
// A full implementation would track the last expression's value.
Ok(Rvalue::Use(Operand::Constant(Constant::Unit)))
}
_ => Err(format!("Unsupported expression: {:?}", expr.kind)),
}
}

/// Helper to lower an expression into an Operand (either a constant or a place).
/// If the expression is complex, it will be assigned to a temporary.
fn lower_expr_to_operand(
&mut self,
expr: &HirExpr,
stmts: &mut Vec<Statement>,
mir: &mut MirBody,
) -> Result<Operand, String> {
let rvalue = self.lower_expr(expr, stmts, mir)?;
// If the rvalue is already a Use(Operand), we can directly return it.
match rvalue {
Rvalue::Use(op) => Ok(op),
_ => {
// Create a temporary and assign the rvalue to it.
let temp = self.new_temp(&expr.span);
stmts.push(Statement::Assign(temp.clone(), rvalue));
Ok(Operand::Move(temp))
}
}
}

/// Create a new temporary place.
fn new_temp(&mut self, span: &Span) -> Place {
let idx = self.next_temp;
self.next_temp += 1;
Place {
name: format!("_t{}", idx),
index: Some(idx + 1000), // high index to distinguish from locals
span: *span,
}
}
}
```

```rust
//! # MIR Instructions
//!
//! Defines all MIR instructions: Place, Operand, Rvalue, Statement, Terminator.

use crate::lexer::Span;
use crate::ast::BinaryOp;

/// A place (local variable or temporary).
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Place {
pub name: String,
pub index: Option<usize>, // None for global/static
pub span: Span,
}

/// An operand (constant or place).
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Operand {
Constant(Constant),
Move(Place),
Copy(Place),
}

/// A constant value.
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Constant {
Int(i64),
Float(f64),
Bool(bool),
String(String),
Unit,
}

/// An R-value (right-hand side of an assignment).
#[derive(Debug, Clone, PartialEq)]
pub enum Rvalue {
Use(Operand),
Binary(BinaryOp, Operand, Operand),
Unary(UnaryOp, Operand),
Call(Operand, Vec<Operand>),
Aggregate(AggregateKind, Vec<Operand>),
}

/// Unary operator.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum UnaryOp {
Neg,
Not,
Deref,
}

/// Aggregate kind.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum AggregateKind {
Tuple,
Array,
Struct(String),
Enum(String, String),
}

/// A statement.
#[derive(Debug, Clone, PartialEq)]
pub enum Statement {
Assign(Place, Rvalue),
StorageLive(Place),
StorageDead(Place),
Return(Option<Operand>), // explicit return statement
Nop,
}

/// Terminator kind.
#[derive(Debug, Clone, PartialEq)]
pub enum TerminatorKind {
Goto(usize),
Switch(Operand, Vec<(Constant, usize)>, usize),
Return,
Call(Operand, Vec<Operand>, usize, usize),
Unreachable,
}

/// A terminator.
#[derive(Debug, Clone, PartialEq)]
pub struct Terminator {
pub kind: TerminatorKind,
pub span: Span,
}
```

```rust
for body in bodies {
self.run(body)?;
```

```rust
//! # Constant Propagation Pass
//!
//! Replaces expressions with constant values where possible.

use super::{MirPass, MirBody};
use super::super::instruction::{Statement, Rvalue, Operand, Constant, UnaryOp, BinaryOp};

pub struct ConstProp;

impl ConstProp {
pub fn new() -> Self {
ConstProp
}

pub fn run_on_body(&self, body: &mut MirBody) {
for block in &mut body.basic_blocks {
for stmt in &mut block.statements {
if let Statement::Assign(_, rvalue) = stmt {
*rvalue = self.fold_rvalue(rvalue);
}
}
}
}

fn fold_rvalue(&self, rvalue: &Rvalue) -> Rvalue {
match rvalue {
Rvalue::Binary(op, op1, op2) => {
if let (Operand::Constant(c1), Operand::Constant(c2)) = (op1, op2) {
match (c1, c2) {
(Constant::Int(a), Constant::Int(b)) => {
let result = match op {
BinaryOp::Add => a + b,
BinaryOp::Sub => a - b,
BinaryOp::Mul => a * b,
BinaryOp::Div => {
if *b != 0 { a / b } else { return rvalue.clone() }
}
BinaryOp::Mod => {
if *b != 0 { a % b } else { return rvalue.clone() }
}
BinaryOp::BitAnd => a & b,
BinaryOp::BitOr => a | b,
BinaryOp::BitXor => a ^ b,
BinaryOp::Shl => a << b,
BinaryOp::Shr => a >> b,
_ => return rvalue.clone(),
};
return Rvalue::Use(Operand::Constant(Constant::Int(result)));
}
(Constant::Float(a), Constant::Float(b)) => {
let result = match op {
BinaryOp::Add => a + b,
BinaryOp::Sub => a - b,
BinaryOp::Mul => a * b,
BinaryOp::Div => {
if *b != 0.0 { a / b } else { return rvalue.clone() }
}
_ => return rvalue.clone(),
};
return Rvalue::Use(Operand::Constant(Constant::Float(result)));
}
(Constant::Bool(a), Constant::Bool(b)) => {
let result = match op {
BinaryOp::And => a & b,
BinaryOp::Or => a | b,
BinaryOp::Eq => a == b,
BinaryOp::Ne => a != b,
_ => return rvalue.clone(),
};
return Rvalue::Use(Operand::Constant(Constant::Bool(result)));
}
_ => return rvalue.clone(),
}
}
rvalue.clone()
}
Rvalue::Unary(op, operand) => {
if let Operand::Constant(c) = operand {
match (op, c) {
(UnaryOp::Neg, Constant::Int(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Int(-v)));
}
(UnaryOp::Neg, Constant::Float(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Float(-v)));
}
(UnaryOp::Not, Constant::Bool(v)) => {
return Rvalue::Use(Operand::Constant(Constant::Bool(!v)));
}
_ => return rvalue.clone(),
}
}
rvalue.clone()
}
_ => rvalue.clone(),
}
}
}

impl MirPass for ConstProp {
fn name(&self) -> &str {
"ConstantPropagation"
}

fn run(&self, body: &mut MirBody) -> Result<(), String> {
self.run_on_body(body);
Ok(())
}
}
```

```rust
//! # CFG Simplification Pass
//!
//! Removes unreachable blocks and simplifies the control-flow graph.

use super::{MirPass, MirBody};
use super::super::instruction::TerminatorKind;
use std::collections::HashSet;

pub struct SimplifyCfg;

impl SimplifyCfg {
pub fn new() -> Self {
SimplifyCfg
}

pub fn run_on_body(&self, body: &mut MirBody) {
let mut reachable = HashSet::new();
let mut worklist = vec![0];

// Mark reachable blocks
while let Some(idx) = worklist.pop() {
if reachable.insert(idx) {
if let Some(block) = body.basic_blocks.get(idx) {
match &block.terminator.kind {
TerminatorKind::Goto(target) => worklist.push(*target),
TerminatorKind::Switch(_, cases, default) => {
for (_, target) in cases {
worklist.push(*target);
}
worklist.push(*default);
}
TerminatorKind::Call(_, _, ret, next) => {
worklist.push(*ret);
worklist.push(*next);
}
_ => {}
}
}
}
}

// Rebuild blocks with only reachable ones
let mut new_blocks = Vec::new();
let mut mapping = vec![usize::MAX; body.basic_blocks.len()];

for i in 0..body.basic_blocks.len() {
if reachable.contains(&i) {
mapping[i] = new_blocks.len();
new_blocks.push(body.basic_blocks[i].clone());
}
}

// Fix terminators to point to remapped blocks
for block in &mut new_blocks {
match &mut block.terminator.kind {
TerminatorKind::Goto(target) => {
if mapping[*target] != usize::MAX {
*target = mapping[*target];
}
}
TerminatorKind::Switch(_, cases, default) => {
for (_, target) in cases {
if mapping[*target] != usize::MAX {
*target = mapping[*target];
}
}
if mapping[*default] != usize::MAX {
*default = mapping[*default];
}
}
TerminatorKind::Call(_, _, ret, next) => {
if mapping[*ret] != usize::MAX {
*ret = mapping[*ret];
}
if mapping[*next] != usize::MAX {
*next = mapping[*next];
}
}
_ => {}
}
}

body.basic_blocks = new_blocks;
}
}

impl MirPass for SimplifyCfg {
fn name(&self) -> &str {
"SimplifyCFG"
}

fn run(&self, body: &mut MirBody) -> Result<(), String> {
self.run_on_body(body);
Ok(())
}
}
```

```rust
//! # Aggressive Dead Code Elimination Pass
//!
//! Removes unused assignments and dead code.

use super::{MirPass, MirBody};
use super::super::instruction::{Statement, Place, Operand, Rvalue, TerminatorKind};
use std::collections::HashSet;

pub struct AggressiveDce;

impl AggressiveDce {
pub fn new() -> Self {
AggressiveDce
}

pub fn run_on_body(&self, body: &mut MirBody) {
let mut used = HashSet::new();

// Collect uses from all statements and terminators
for block in &body.basic_blocks {
for stmt in &block.statements {
match stmt {
Statement::Assign(_, rvalue) => {
self.collect_operands(rvalue, &mut used);
}
Statement::Return(Some(op)) => self.collect_operand(op, &mut used),
Statement::StorageLive(_) | Statement::StorageDead(_) => {}
_ => {}
}
}
// Collect operands from terminator
self.collect_terminator_operands(&block.terminator, &mut used);
}

// Remove unused assignments, keeping all non-assign statements
for block in &mut body.basic_blocks {
block.statements.retain(|stmt| {
if let Statement::Assign(place, _) = stmt {
self.is_place_used(place, &used)
} else if let Statement::Return(Some(op)) = stmt {
true // return is always live
} else {
true // keep other statements
}
});
}
}

fn collect_operands(&self, rvalue: &Rvalue, used: &mut HashSet<Place>) {
match rvalue {
Rvalue::Use(op) => self.collect_operand(op, used),
Rvalue::Binary(_, op1, op2) => {
self.collect_operand(op1, used);
self.collect_operand(op2, used);
}
Rvalue::Unary(_, op) => self.collect_operand(op, used),
Rvalue::Call(func, args) => {
self.collect_operand(func, used);
for arg in args {
self.collect_operand(arg, used);
}
}
Rvalue::Aggregate(_, ops) => {
for op in ops {
self.collect_operand(op, used);
}
}
}
}

fn collect_operand(&self, op: &Operand, used: &mut HashSet<Place>) {
match op {
Operand::Move(p) | Operand::Copy(p) => {
self.add_place_with_prefixes(p, used);
}
_ => {}
}
}

fn collect_terminator_operands(&self, terminator: &super::super::instruction::Terminator, used: &mut HashSet<Place>) {
match &terminator.kind {
TerminatorKind::Switch(op, _, _) => self.collect_operand(op, used),
TerminatorKind::Call(func, args, _, _) => {
self.collect_operand(func, used);
for arg in args {
self.collect_operand(arg, used);
}
}
_ => {}
}
}

fn add_place_with_prefixes(&self, place: &Place, set: &mut HashSet<Place>) {
set.insert(place.clone());
if place.index.is_some() {
let mut prefix = place.clone();
prefix.index = None;
set.insert(prefix);
}
}

fn is_place_used(&self, place: &Place, used: &HashSet<Place>) -> bool {
if used.contains(place) {
return true;
}
if place.index.is_some() {
let mut prefix = place.clone();
prefix.index = None;
return used.contains(&prefix);
}
false
}
}

impl MirPass for AggressiveDce {
fn name(&self) -> &str {
```

```rust
//! # MIR Optimization Tests
//!
//! Tests the MIR optimization passes.

#[cfg(test)]
mod tests {
use crate::lexer::Tokenizer;
use crate::parser::Parser;
use crate::sema::Sema;
use crate::hir::Lowering;
use crate::mir::MirBuilder;
use crate::mir::passes::{ConstProp, SimplifyCfg, AggressiveDce, PassManager};
use crate::diagnostics::CollectorEmitter;

#[test]
fn test_mir_optimization() {
let src = "fn main() -> int { let x = 1 + 2; return x; }";
let mut tok = Tokenizer::new(src);
let mut parser = Parser::new(src, &mut tok);
let file = parser.parse_program();
assert!(!parser.has_errors());

let mut emitter = CollectorEmitter::new();
let mut sema = Sema::new(&mut emitter);
sema.analyze(&file).unwrap();

let sema_ctx = sema.context();
let mut lowering = Lowering::new(sema_ctx);
let hir_file = lowering.lower_file(&file).unwrap();

let mut builder = MirBuilder::new();
let mut mir_bodies = builder.build(&hir_file).unwrap();
let mut main_body = mir_bodies.remove(0);

// Run optimizations
let mut pm = PassManager::new();
pm.add_pass(Box::new(ConstProp::new()));
pm.add_pass(Box::new(SimplifyCfg::new()));
pm.add_pass(Box::new(AggressiveDce::new()));

let result = pm.run(&mut main_body);
assert!(result.is_ok());

// Check that optimization didn't break anything
assert!(!main_body.is_empty());
}
}
```
