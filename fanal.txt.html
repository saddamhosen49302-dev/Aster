📁 scripts/optimize.sh (অটোমেটিক অপটিমাইজেশন স্ক্রিপ্ট)

```bash
#!/bin/bash
# Aster Full System Optimizer v1.0
# Runs all optimizations: LTO, strip, UPX, parallel build, and cache clear

set -e

echo "🚀 Starting Aster Full System Optimization..."

# 1. Clean old builds
echo "🧹 Cleaning old builds..."
cargo clean

# 2. Set release profile optimizations
echo "⚙️ Setting release profile optimizations..."
cat > .cargo/config.toml << EOF
[build]
rustflags = ["-C", "link-arg=-fuse-ld=lld", "-C", "target-cpu=native"]

[profile.release]
lto = true
codegen-units = 1
panic = "abort"
strip = true
opt-level = 3
EOF

# 3. Build with parallel codegen
echo "📦 Building with parallel codegen (Rayon)..."
cargo build --release --workspace

# 4. Strip binaries
echo "✂️ Stripping binaries..."
strip target/release/asterc 2>/dev/null || true
strip target/release/multilang 2>/dev/null || true
strip target/release/asterpm 2>/dev/null || true

# 5. UPX compress (optional)
if command -v upx &> /dev/null; then
echo "📦 Compressing with UPX..."
upx --best --lzma target/release/asterc 2>/dev/null || true
upx --best --lzma target/release/multilang 2>/dev/null || true
upx --best --lzma target/release/asterpm 2>/dev/null || true
else
echo "⚠️ UPX not installed. Skipping compression."
fi

# 6. Run tests to verify
echo "🧪 Running tests to verify optimizations..."
cargo test --workspace --release

# 7. Show sizes
echo ""
echo "📊 Final Binary Sizes:"
ls -lh target/release/asterc target/release/multilang target/release/asterpm

echo ""
echo "✅ Optimization complete! Your Aster is now faster, smaller, and production-ready."
```
