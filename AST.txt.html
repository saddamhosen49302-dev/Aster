//! # Hardware Type Utilities
//!
//! Defines hardware type information and utilities.

use crate::lexer::Span;
use serde::{Deserialize, Serialize};

/// Hardware type representation – now with `Span` for better error reporting.
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub enum HwType {
Logic { width: Option<usize>, span: Span },
Bit { width: Option<usize>, span: Span },
Int { width: usize, signed: bool, span: Span },
Struct(Vec<(String, HwType)>, Span),
Enum(Vec<String>, Span),
Array(Box<HwType>, usize, Span),
}

impl HwType {
/// Returns the bit width of the type if fixed.
pub fn bit_width(&self) -> Option<usize> {
match self {
HwType::Logic { width, .. } => *width,
HwType::Bit { width, .. } => *width,
HwType::Int { width, .. } => Some(*width),
HwType::Array(elem, size, ..) => elem.bit_width().map(|w| w * size),
_ => None,
}
}

/// Returns true if the type is signed.
pub fn is_signed(&self) -> bool {
matches!(self, HwType::Int { signed: true, .. })
}

/// Converts the type to a SystemVerilog string representation.
pub fn to_verilog(&self) -> String {
match self {
HwType::Logic { width, .. } => {
if let Some(w) = width {
format!("logic [{}:0]", w - 1)
} else {
"logic".to_string()
}
}
HwType::Bit { width, .. } => {
if let Some(w) = width {
format!("bit [{}:0]", w - 1)
} else {
"bit".to_string()
}
}
HwType::Int { width, signed, .. } => {
let sign = if *signed { "signed " } else { "" };
// SystemVerilog standard: use logic/bit with signed qualifier
format!("{}logic [{}:0]", sign, width - 1)
}
HwType::Struct(fields, ..) => {
let parts: Vec<String> = fields
.iter()
.map(|(n, t)| format!("{}: {}", n, t.to_verilog()))
.collect();
format!("struct {{{}}}", parts.join(", "))
}
HwType::Enum(variants, ..) => format!("enum {{{}}}", variants.join(", ")),
HwType::Array(elem, size, ..) => format!("{} [{}]", elem.to_verilog(), size),
}
}
}

//! # Visitor Pattern
//!
//! Allows walking the AST without modifying it.

use super::*;

pub trait Visitor<'ast> {
type Result;
type Error;

fn visit_file(&mut self, f: &'ast File) -> Result<Self::Result, Self::Error>;
fn visit_item(&mut self, i: &'ast Item) -> Result<Self::Result, Self::Error>;
fn visit_block(&mut self, b: &'ast Block) -> Result<Self::Result, Self::Error>;
fn visit_stmt(&mut self, s: &'ast Stmt) -> Result<Self::Result, Self::Error>;
fn visit_expr(&mut self, e: &'ast Expr) -> Result<Self::Result, Self::Error>;
fn visit_ty(&mut self, t: &'ast Ty) -> Result<Self::Result, Self::Error>;
fn visit_pat(&mut self, p: &'ast Pat) -> Result<Self::Result, Self::Error>;
}

pub mod walk {
use super::*;

pub fn walk_file<'ast, V: Visitor<'ast>>(v: &mut V, f: &'ast File) -> Result<V::Result, V::Error> {
for item in &f.items {
walk_item(v, item)?;
}
v.visit_file(f)
}

pub fn walk_item<'ast, V: Visitor<'ast>>(v: &mut V, i: &'ast Item) -> Result<V::Result, V::Error> {
match i {
Item::Fn(sig) => {
for (pat, ty) in &sig.args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = &sig.ret_ty {
v.visit_ty(ty)?;
}
walk_block(v, &sig.body)?;
}
Item::Struct(_, fields, _, _, _) => {
for f in fields {
v.visit_ty(&f.ty)?;
}
}
Item::Enum(_, variants, _, _, _) => {
for var in variants {
for ty in &var.fields {
v.visit_ty(ty)?;
}
}
}
Item::Impl(ty, _, _, _, body, _) => {
v.visit_ty(ty)?;
walk_block(v, body)?;
}
Item::Trait(_, _, _, items, _) => {
for item in items {
match item {
TraitItem::Fn(_, args, ret_ty, body, _) => {
for (pat, ty) in args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = ret_ty {
v.visit_ty(ty)?;
}
if let Some(b) = body {
walk_block(v, b)?;
}
}
TraitItem::Type(_, _) => {}
TraitItem::Const(_, ty, _) => v.visit_ty(ty)?,
_ => {}
}
}
}
Item::Extern(_, items, _) => {
for item in items {
match item {
ExternItem::Fn(_, args, ret_ty, _) => {
for (pat, ty) in args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = ret_ty {
v.visit_ty(ty)?;
}
}
ExternItem::Static(_, ty, _, _) => v.visit_ty(ty)?,
_ => {}
}
}
}
// ✅ FIX: added missing hardware arms
Item::HwModule(m) => {
for stmt in &m.body {
walk_stmt(v, stmt)?;
}
}
Item::HwInterface(_, _, body, _) => {
for stmt in body {
walk_stmt(v, stmt)?;
}
}
_ => {}
}
v.visit_item(i)
}

pub fn walk_block<'ast, V: Visitor<'ast>>(v: &mut V, b: &'ast Block) -> Result<V::Result, V::Error> {
for stmt in &b.stmts {
walk_stmt(v, stmt)?;
}
v.visit_block(b)
}

pub fn walk_stmt<'ast, V: Visitor<'ast>>(v: &mut V, s: &'ast Stmt) -> Result<V::Result, V::Error> {
match s {
Stmt::Let(pat, ty, expr, _) => {
walk_pat(v, pat)?;
if let Some(ty) = ty {
v.visit_ty(ty)?;
}
walk_expr(v, expr)?;
}
Stmt::Expr(expr, _) | Stmt::Semi(expr, _) => walk_expr(v, expr)?,
Stmt::Return(expr, _) => {
if let Some(e) = expr {
walk_expr(v, e)?;
}
}
Stmt::If(cond, then_b, else_b, _) => {
walk_expr(v, cond)?;
walk_block(v, then_b)?;
if let Some(e) = else_b {
walk_stmt(v, e)?;
}
}
Stmt::While(cond, body, _) => {
walk_expr(v, cond)?;
walk_block(v, body)?;
}
Stmt::For(pat, iter, body, _) => {
walk_pat(v, pat)?;
walk_expr(v, iter)?;
walk_block(v, body)?;
}
Stmt::Loop(_, body, _) => walk_block(v, body)?,
Stmt::Break(expr, _) => {
if let Some(e) = expr {
walk_expr(v, e)?;
}
}
Stmt::Match(expr, arms, _) => {
walk_expr(v, expr)?;
for arm in arms {
walk_pat(v, &arm.pat)?;
if let Some(guard) = &arm.guard {
walk_expr(v, guard)?;
}
walk_block(v, &arm.body)?;
}
}
Stmt::Try(try_b, catches, _) => {
walk_block(v, try_b)?;
for (pat, body) in catches {
walk_pat(v, pat)?;
walk_block(v, body)?;
}
}
Stmt::Block(block, _) => walk_block(v, block)?,
Stmt::HwAssign(lhs, rhs, _) => {
walk_expr(v, lhs)?;
walk_expr(v, rhs)?;
}
Stmt::HwAlways(_, body, _) => walk_block(v, body)?,
Stmt::HwAssert(expr, _) => walk_expr(v, expr)?,
_ => {}
}
v.visit_stmt(s)
}

pub fn walk_expr<'ast, V: Visitor<'ast>>(v: &mut V, e: &'ast Expr) -> Result<V::Result, V::Error> {
match e {
Expr::Binary(l, _, r, _) => {
walk_expr(v, l)?;
walk_expr(v, r)?;
}
Expr::Unary(_, e2, _) => walk_expr(v, e2)?,
Expr::Call(func, args, _) => {
walk_expr(v, func)?;
for arg in args {
walk_expr(v, arg)?;
}
}
Expr::Block(b, _) => walk_block(v, b)?,
Expr::Paren(e2, _) => walk_expr(v, e2)?,
_ => {}
}
v.visit_expr(e)
}

pub fn walk_pat<'ast, V: Visitor<'ast>>(v: &mut V, p: &'ast Pat) -> Result<V::Result, V::Error> {
match p {
Pat::Tuple(pats, _) => {
for p2 in pats {
walk_pat(v, p2)?;
}
}
Pat::Struct(fields, _) => {
for (_, p2) in fields {
walk_pat(v, p2)?;
}
}
_ => {}
}
v.visit_pat(p)
}
}

//! # Visitor Pattern
//!
//! Allows walking the AST without modifying it.

use super::*;

pub trait Visitor<'ast> {
type Result;
type Error;

fn visit_file(&mut self, f: &'ast File) -> Result<Self::Result, Self::Error>;
fn visit_item(&mut self, i: &'ast Item) -> Result<Self::Result, Self::Error>;
fn visit_block(&mut self, b: &'ast Block) -> Result<Self::Result, Self::Error>;
fn visit_stmt(&mut self, s: &'ast Stmt) -> Result<Self::Result, Self::Error>;
fn visit_expr(&mut self, e: &'ast Expr) -> Result<Self::Result, Self::Error>;
fn visit_ty(&mut self, t: &'ast Ty) -> Result<Self::Result, Self::Error>;
fn visit_pat(&mut self, p: &'ast Pat) -> Result<Self::Result, Self::Error>;
}

pub mod walk {
use super::*;

pub fn walk_file<'ast, V: Visitor<'ast>>(v: &mut V, f: &'ast File) -> Result<V::Result, V::Error> {
for item in &f.items {
walk_item(v, item)?;
}
v.visit_file(f)
}

pub fn walk_item<'ast, V: Visitor<'ast>>(v: &mut V, i: &'ast Item) -> Result<V::Result, V::Error> {
match i {
Item::Fn(sig) => {
for (pat, ty) in &sig.args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = &sig.ret_ty {
v.visit_ty(ty)?;
}
walk_block(v, &sig.body)?;
}
// ✅ FIXED: Impl now uses ImplDef
Item::Impl(def) => {
v.visit_ty(&def.ty)?;
for item in &def.items {
walk_impl_item(v, item)?;
}
}
Item::Struct(_, fields, _, _, _) => {
for f in fields {
v.visit_ty(&f.ty)?;
}
}
Item::Enum(_, variants, _, _, _) => {
for var in variants {
for ty in &var.fields {
v.visit_ty(ty)?;
}
}
}
Item::Trait(_, _, _, items, _) => {
for item in items {
match item {
TraitItem::Fn(_, args, ret_ty, body, _) => {
for (pat, ty) in args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = ret_ty {
v.visit_ty(ty)?;
}
if let Some(b) = body {
walk_block(v, b)?;
}
}
TraitItem::Type(_, _) => {}
TraitItem::Const(_, ty, _) => v.visit_ty(ty)?,
_ => {}
}
}
}
Item::Extern(_, items, _) => {
for item in items {
match item {
ExternItem::Fn(_, args, ret_ty, _) => {
for (pat, ty) in args {
walk_pat(v, pat)?;
v.visit_ty(ty)?;
}
if let Some(ty) = ret_ty {
v.visit_ty(ty)?;
}
}
ExternItem::Static(_, ty, _, _) => v.visit_ty(ty)?,
_ => {}
}

pub fn walk_impl_item<'ast, V: Visitor<'ast>>(

) -> Result<V::Result, V::Error> {

}
ImplItem::Type(_, ty, _) => {
v.visit_ty(ty)?;
}

pub fn walk_block<'ast, V: Visitor<'ast>>(v: &mut V, b: &'ast Block) -> Result<V::Result, V::Error> {
for stmt in &b.stmts {
walk_stmt(v, stmt)?;
}
v.visit_block(b)
}

pub fn walk_stmt<'ast, V: Visitor<'ast>>(v: &mut V, s: &'ast Stmt) -> Result<V::Result, V::Error> {
match s {
Stmt::Let(pat, ty, expr, _) => {
walk_pat(v, pat)?;
if let Some(ty) = ty {
v.visit_ty(ty)?;
}
walk_expr(v, expr)?;
}
Stmt::Expr(expr, _) | Stmt::Semi(expr, _) => walk_expr(v, expr)?,
Stmt::Return(expr, _) => {
if let Some(e) = expr {
walk_expr(v, e)?;
}
}
Stmt::If(cond, then_b, else_b, _) => {
walk_expr(v, cond)?;
walk_block(v, then_b)?;
if let Some(e) = else_b {
walk_stmt(v, e)?;
}
}
Stmt::While(cond, body, _) => {
walk_expr(v, cond)?;
walk_block(v, body)?;
}
Stmt::For(pat, iter, body, _) => {
walk_pat(v, pat)?;
walk_expr(v, iter)?;
walk_block(v, body)?;
}
Stmt::Loop(_, body, _) => walk_block(v, body)?,
Stmt::Break(expr, _) => {
if let Some(e) = expr {
walk_expr(v, e)?;
}
}
Stmt::Match(expr, arms, _) => {
walk_expr(v, expr)?;
for arm in arms {
walk_pat(v, &arm.pat)?;
if let Some(guard) = &arm.guard {
walk_expr(v, guard)?;
}
walk_block(v, &arm.body)?;
}
}
Stmt::Try(try_b, catches, _) => {
walk_block(v, try_b)?;
for (pat, body) in catches {
walk_pat(v, pat)?;
walk_block(v, body)?;
}
}
Stmt::Block(block, _) => walk_block(v, block)?,
Stmt::HwAssign(lhs, rhs, _) => {
walk_expr(v, lhs)?;
walk_expr(v, rhs)?;
}
Stmt::HwAlways(_, body, _) => walk_block(v, body)?,
Stmt::HwAssert(expr, _) => walk_expr(v, expr)?,
_ => {}
}
v.visit_stmt(s)
}

pub fn walk_expr<'ast, V: Visitor<'ast>>(v: &mut V, e: &'ast Expr) -> Result<V::Result, V::Error> {
match e {
Expr::Binary(l, _, r, _) => {
walk_expr(v, l)?;
walk_expr(v, r)?;
}
Expr::Unary(_, e2, _) => walk_expr(v, e2)?,
Expr::Call(func, args, _) => {
walk_expr(v, func)?;
for arg in args {
walk_expr(v, arg)?;
}
}
Expr::Block(b, _) => walk_block(v, b)?,
Expr::Paren(e2, _) => walk_expr(v, e2)?,
_ => {}
}
v.visit_expr(e)
}

pub fn walk_pat<'ast, V: Visitor<'ast>>(v: &mut V, p: &'ast Pat) -> Result<V::Result, V::Error> {
match p {
Pat::Tuple(pats, _) => {
for p2 in pats {
walk_pat(v, p2)?;
}
}
Pat::Struct(fields, _) => {
for (_, p2) in fields {
walk_pat(v, p2)?;
}
}
_ => {}
}
v.visit_pat(p)
}
}
